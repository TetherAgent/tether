    ```dart
    class ReplayScreen extends StatefulWidget {
      const ReplayScreen({super.key});
      @override
      State&lt;ReplayScreen&gt; createState() => _ReplayScreenState();
    }

    class _ReplayScreenState extends State&lt;ReplayScreen&gt; {
      late RelayClient _relay;
      late Terminal _terminal;
      StreamSubscription&lt;ReplayOutput&gt;? _replaySub;
      String? _sessionId;

      @override
      void didChangeDependencies() {
        super.didChangeDependencies();
        _relay = Provider.of&lt;RelayClient&gt;(context, listen: false);
        _sessionId = GoRouterState.of(context).pathParameters['id'];
        if (_replaySub == null &amp;&amp; _sessionId != null) {
          _replaySub = _relay.replayOutputStream
            .where((ro) => ro.sessionId == _sessionId)
            .listen((ro) => _terminal.write(ro.data));
          WidgetsBinding.instance.addPostFrameCallback((_) {
            _relay.sendSubscribe(_sessionId!, mode: 'observe', after: 0);
          });
        }
      }

      @override
      void dispose() {
        _replaySub?.cancel();
        super.dispose();
      }

      @override
      Widget build(BuildContext context) {
        final l = AppLocalizations.of(context)!;
        final session = _relay.historySessions
          .firstWhereOrNull((s) => s.id == _sessionId);
        final title = session?.title ?? session?.provider ?? _sessionId ?? '';
        return Scaffold(
          appBar: AppBar(
            title: Text(title),
            actions: [Chip(label: Text(l.replay))],
          ),
          body: TerminalView(_terminal),  // xterm TerminalView, read-only (no input)
        );
      }
    }
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-05-01 | Spoofing | ChatSessionSurface | accept | Session ownership validated by RelayClient subscribeFrame (Plan 02); UI layer does not re-validate |
| T-09-05-02 | Information Disclosure | AssistantBubble flutter_markdown | mitigate | Use MarkdownBody from flutter_markdown without HTML mode (selectable: false, data only); no JS execution path |
| T-09-05-03 | Denial of Service | replayOutputStream terminal write | mitigate | Cap replay at 5000 events: cancel _replaySub on ReplayDone frame; xterm pub.dev has internal buffer limits |
| T-09-05-04 | Elevation of Privilege | SelectOptionsRow sendMessage | accept | Options sourced from authenticated relay events via ConversationService; no additional server-side exposure from client selection |
