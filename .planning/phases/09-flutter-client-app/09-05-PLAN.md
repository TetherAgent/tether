**创建 native/flutter/lib/widgets/keyboard_toolbar.dart**

KeyboardToolbar 设计为纯 UI 组件，通过回调与 TerminalScreen 解耦：

**重要：State 类命名为 `KeyboardToolbarState`（公开，非私有）**，因为 TerminalScreen 需要通过
`GlobalKey<KeyboardToolbarState>` 跨文件访问其 `isCtrlHeld` 和 `resetCtrl()` 成员。
Dart 中私有类（`_KeyboardToolbarState`）无法在文件外通过 GlobalKey 泛型参数引用。

```dart
// lib/widgets/keyboard_toolbar.dart
// UI-SPEC Screen 4 键盘工具栏规范:
//   48dp 高度，tetherSurface 背景，3 个等宽按钮: Ctrl / Esc / Tab
//   Ctrl: hold 状态 → tetherAccent 20% 背景 + 1px tetherAccent 边框
//   观察模式: 全部按钮 opacity 0.4，禁用
// 使用 MediaQuery.viewInsets.bottom > 0 判断键盘是否可见（在 TerminalScreen 中决定）
// 不使用 keyboard_actions 插件（OHOS 不支持 — RESEARCH.md Anti-Patterns）
import 'dart:async';
import 'package:flutter/material.dart';
import '../theme.dart';

class KeyboardToolbar extends StatefulWidget {
  /// 观察模式时禁用所有按钮（D-21）
  final bool isObserveMode;
  /// 发送字节的回调（由 TerminalScreen 提供，转发到 relay/lan client）
  final void Function(String data) onSend;

  const KeyboardToolbar({super.key, required this.onSend, this.isObserveMode = false});

  @override
  State<KeyboardToolbar> createState() => KeyboardToolbarState();
}

// 公开 State 类（非私有）— TerminalScreen 跨文件通过 GlobalKey<KeyboardToolbarState> 访问
class KeyboardToolbarState extends State<KeyboardToolbar> {
  bool _ctrlHeld = false;
  Timer? _ctrlTimeout;

  // Ctrl hold 状态机
  // 镜像 RESEARCH.md Pitfall 5 + PATTERNS.md terminal_screen Ctrl modifier 模式
  void _onCtrlTap() {
    if (widget.isObserveMode) return;
    _ctrlTimeout?.cancel();
    setState(() => _ctrlHeld = !_ctrlHeld);
    if (_ctrlHeld) {
      // 3 秒后自动重置（防止误操作锁定）
      _ctrlTimeout = Timer(const Duration(seconds: 3), () {
        if (mounted) setState(() => _ctrlHeld = false);
      });
    }
  }

  /// 当 Ctrl 处于 hold 状态时，下一个字符经过 ASCII control 变换
  /// 调用方（TerminalScreen）在 terminal.onOutput 中使用此方法
  bool get isCtrlHeld => _ctrlHeld;

  void resetCtrl() {
    _ctrlTimeout?.cancel();
    if (mounted) setState(() => _ctrlHeld = false);
  }

  void _onEscTap() {
    if (widget.isObserveMode) return;
    widget.onSend(String.fromCharCode(0x1B)); // ESC
  }

  void _onTabTap() {
    if (widget.isObserveMode) return;
    widget.onSend(String.fromCharCode(0x09)); // Tab
  }

  @override
  void dispose() {
    _ctrlTimeout?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Opacity(
      opacity: widget.isObserveMode ? 0.4 : 1.0,
      child: Container(
        height: 48,
        decoration: BoxDecoration(
          color: tetherSurface,
          border: const Border(top: BorderSide(color: tetherBorder)),
        ),
        child: Row(
          children: [
            _ToolbarButton(
              label: 'Ctrl',
              isActive: _ctrlHeld,
              onTap: _onCtrlTap,
            ),
            _ToolbarButton(label: 'Esc', onTap: _onEscTap),
            _ToolbarButton(label: 'Tab', onTap: _onTabTap),
          ],
        ),
      ),
    );
  }
}

class _ToolbarButton extends StatelessWidget {
  final String label;
  final bool isActive;
  final VoidCallback onTap;

  const _ToolbarButton({required this.label, required this.onTap, this.isActive = false});

  @override
  Widget build(BuildContext context) {
    return Expanded(
      child: GestureDetector(
        onTap: onTap,
        child: Semantics(
          label: '$label 键',
          child: Container(
            alignment: Alignment.center,
            decoration: isActive
                ? BoxDecoration(
                    color: tetherAccent.withValues(alpha: 0.2),
                    border: Border.all(color: tetherAccent, width: 1),
                  )
                : null,
            child: Text(
              label,
              style: const TextStyle(fontSize: 12, color: tetherForeground),
            ),
          ),
        ),
      ),
    );
  }
}
```

关键实现要求：

```dart
// lib/screens/terminal_screen.dart
// UI-SPEC Screen 4 — 终端控制屏幕
// D-04/05: xterm TerminalView，tetherTerminalTheme，支持 alternate screen TUI
// D-12/13: KeyboardToolbar（Ctrl/Esc/Tab），键盘可见时显示
// D-14: pinch-to-zoom，fontSize clamp(10, 24)
// D-15: 旋转 → terminal.onResize → sendResize
// D-21: 模式切换 chip（控制/观察），观察模式 toolbar dimmed
// D-22/23/24: 指数退避重连；后台返回重连；latestEventId 游标
import 'dart:async';
import 'dart:convert';
import 'package:flutter/material.dart';
import 'package:flutter_gen/gen_l10n/app_localizations.dart';
import 'package:provider/provider.dart';
import 'package:xterm/xterm.dart';
import '../models/protocol.dart';
import '../services/relay_client.dart';
import '../theme.dart';
import '../widgets/keyboard_toolbar.dart';

class TerminalScreen extends StatefulWidget {
  final String sessionId;
  const TerminalScreen({super.key, required this.sessionId});
  @override
  State<TerminalScreen> createState() => _TerminalScreenState();
}

class _TerminalScreenState extends State<TerminalScreen>
    with WidgetsBindingObserver {
  final Terminal _terminal = Terminal();
  // GlobalKey 使用公开的 KeyboardToolbarState（非私有 _KeyboardToolbarState）
  // 跨文件 GlobalKey 泛型必须引用公开类，否则 Dart 编译错误
  final GlobalKey<KeyboardToolbarState> _toolbarKey = GlobalKey();
  double _fontSize = 13.0;
  double _scaleStart = 1.0;
  RelayClientMode _mode = RelayClientMode.control;
  bool _subscribed = false;
  // StreamSubscription 保存引用以在 dispose() 中取消，防止泄漏
  StreamSubscription<Event>? _eventSub;

  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addObserver(this); // D-23

    // terminal.onOutput: 用户键盘输入 → 转发到 relay
    // Ctrl modifier: 由 KeyboardToolbar 的 isCtrlHeld 状态处理
    _terminal.onOutput = (data) {
      final toolbarState = _toolbarKey.currentState;
      if (_mode == RelayClientMode.observe) {
        // D-21: 观察模式不发送输入
        _showObserveModeSnack();
        return;
      }
      final relay = context.read<RelayClient>();
      // Ctrl hold + 0x1F 变换 (RESEARCH.md Pitfall 5)
      if (toolbarState != null && toolbarState.isCtrlHeld) {
        toolbarState.resetCtrl();
        if (data.isNotEmpty) {
          final controlByte = data.codeUnitAt(0) & 0x1F;
          relay.sendInput(widget.sessionId, String.fromCharCode(controlByte));
          return;
        }
      }
      relay.sendInput(widget.sessionId, data);
    };

    // terminal.onResize: 布局变化 → sendResize (D-15)
    _terminal.onResize = (cols, rows, pixelWidth, pixelHeight) {
      if (_subscribed) {
        context.read<RelayClient>().sendResize(widget.sessionId, cols, rows);
      }
    };

    // 延迟订阅，确保 context 已可用
    WidgetsBinding.instance.addPostFrameCallback((_) => _subscribe());
  }

  void _subscribe() {
    final relay = context.read<RelayClient>();
    relay.subscribe(widget.sessionId, _mode);
    _subscribed = true;
    // 监听 relay events 并写入 terminal
    // terminal.output payload: { "data": "<base64 PTY bytes>" }
    // 确认自 apps/web/src/components/session/session-surface.tsx line 530-535
    _eventSub = relay.eventStream.listen((event) {
      if (event.event.sessionId == widget.sessionId) {
        final raw = event.event.payload['data'] as String?;
        if (raw != null) {
          try {
            final bytes = base64.decode(raw);
            _terminal.write(String.fromCharCodes(bytes));
          } catch (_) {
```
