// lib/models/protocol.dart
// Dart 3 sealed class 协议类型 — 镜像 packages/protocol/src/index.ts
// 手写版本，生成脚本见 packages/protocol/scripts/gen-dart.sh
// 格式：每个 sealed class 父类有 fromJson 工厂方法；子类 final class 有 toJson

enum RelaySessionStatus {
  running, stopped, completed, failed, lost;
  
  static RelaySessionStatus fromString(String s) => switch (s) {
    'running'   => RelaySessionStatus.running,
    'stopped'   => RelaySessionStatus.stopped,
    'completed' => RelaySessionStatus.completed,
    'failed'    => RelaySessionStatus.failed,
    'lost'      => RelaySessionStatus.lost,
    _           => RelaySessionStatus.lost,
  };
  
  String toJson() => name;
}

enum RelayClientMode {
  control, observe;
  
  static RelayClientMode fromString(String s) => switch (s) {
    'control' => RelayClientMode.control,
    'observe' => RelayClientMode.observe,
    _         => RelayClientMode.control,
  };
  
  String toJson() => name;
}

final class RelaySession {
  final String id;
  final String provider;
  final String title;
  final String projectPath;
  final String? accountId;
  final String? workspaceId;
  final String? gatewayId;
  final String? userId;
  final RelaySessionStatus status;
  final String transport;
  final int lastActiveAt;
  
  const RelaySession({
    required this.id, required this.provider, required this.title,
    required this.projectPath, required this.status, required this.transport,
    required this.lastActiveAt, this.accountId, this.workspaceId,
    this.gatewayId, this.userId,
  });
  
  factory RelaySession.fromJson(Map<String, dynamic> json) => RelaySession(
    id: json['id'] as String,
    provider: json['provider'] as String,
    title: json['title'] as String,
    projectPath: json['projectPath'] as String,
    status: RelaySessionStatus.fromString(json['status'] as String),
    transport: json['transport'] as String,
    lastActiveAt: json['lastActiveAt'] as int,
    accountId: json['accountId'] as String?,
    workspaceId: json['workspaceId'] as String?,
    gatewayId: json['gatewayId'] as String?,
    userId: json['userId'] as String?,
  );
}

final class RelayTerminalEvent {
  final int id;
  final String sessionId;
  final String type;
  final int ts;
  final Map<String, dynamic> payload;
  
  const RelayTerminalEvent({
    required this.id, required this.sessionId, required this.type,
    required this.ts, required this.payload,
  });
  
  factory RelayTerminalEvent.fromJson(Map<String, dynamic> json) => RelayTerminalEvent(

    sessionId: json['sessionId'] as String,
    type: json['type'] as String,
    ts: json['ts'] as int,
    payload: Map<String, dynamic>.from(json['payload'] as Map),
  );

// ---- RelayClientToServerFrame — 7 variants ----

sealed class RelayClientToServerFrame {
  const RelayClientToServerFrame();
  Map<String, dynamic> toJson();
}

final class ClientAuth extends RelayClientToServerFrame {
  final String? token;
  final String? secret;
  const ClientAuth({this.token, this.secret});
  @override
  Map<String, dynamic> toJson() => {
    'type': 'client.auth',
    if (token != null) 'token': token,
    if (secret != null) 'secret': secret,
  };
}

final class ClientList extends RelayClientToServerFrame {
  const ClientList();
  @override
  Map<String, dynamic> toJson() => {'type': 'client.list'};
}

final class ClientSubscribe extends RelayClientToServerFrame {
  final String sessionId;
  final int? after;
  final int? tail;
  final RelayClientMode mode;
  const ClientSubscribe({required this.sessionId, required this.mode, this.after, this.tail});
  @override
  Map<String, dynamic> toJson() => {
    'type': 'client.subscribe',
    'sessionId': sessionId,
    'mode': mode.toJson(),
    if (after != null) 'after': after,
    if (tail != null) 'tail': tail,
  };
}

final class ClientInput extends RelayClientToServerFrame {
  final String sessionId;
  final String data;
  const ClientInput({required this.sessionId, required this.data});
  @override
  Map<String, dynamic> toJson() => {'type': 'client.input', 'sessionId': sessionId, 'data': data};
}

final class ClientResize extends RelayClientToServerFrame {
  final String sessionId;
  final int cols;
  final int rows;
  const ClientResize({required this.sessionId, required this.cols, required this.rows});
  @override
  Map<String, dynamic> toJson() => {
    'type': 'client.resize', 'sessionId': sessionId, 'cols': cols, 'rows': rows
  };
}

final class ClientStop extends RelayClientToServerFrame {
  final String sessionId;
  const ClientStop({required this.sessionId});
  @override
  Map<String, dynamic> toJson() => {'type': 'client.stop', 'sessionId': sessionId};
}

final class ClientDetach extends RelayClientToServerFrame {
  final String sessionId;
  const ClientDetach({required this.sessionId});
  @override
  Map<String, dynamic> toJson() => {'type': 'client.detach', 'sessionId': sessionId};
}

// ---- RelayServerToClientFrame — 7 variants ----

sealed class RelayServerToClientFrame {
  const RelayServerToClientFrame();
  
  factory RelayServerToClientFrame.fromJson(Map<String, dynamic> json) {
    return switch (json['type'] as String) {
      'client.auth.ok'     => ClientAuthOk(clientId: json['clientId'] as String),
      'client.auth.failed' => ClientAuthFailed(
          code: json['code'] as String, message: json['message'] as String),
      'sessions'           => Sessions(
          sessions: (json['sessions'] as List)
              .map((e) => RelaySession.fromJson(e as Map<String, dynamic>))
              .toList()),
      'hello'              => Hello(
          clientId: json['clientId'] as String,
          gatewayId: json['gatewayId'] as String?),
      'event'              => Event(
          event: RelayTerminalEvent.fromJson(json['event'] as Map<String, dynamic>)),
      'replay.done'        => ReplayDone(
          sessionId: json['sessionId'] as String,
          latestEventId: json['latestEventId'] as int),
      'error'              => TetherError(
          code: json['code'] as String,
          message: json['message'] as String,
          sessionId: json['sessionId'] as String?),
      final t              => throw FormatException('Unknown Relay frame type: $t'),
    };
  }
}

final class ClientAuthOk extends RelayServerToClientFrame {
  final String clientId;
  const ClientAuthOk({required this.clientId});
}

final class ClientAuthFailed extends RelayServerToClientFrame {
  final String code;
  final String message;
  const ClientAuthFailed({required this.code, required this.message});
}

final class Sessions extends RelayServerToClientFrame {
  final List<RelaySession> sessions;
  const Sessions({required this.sessions});
}

final class Hello extends RelayServerToClientFrame {
  final String clientId;
  final String? gatewayId;
  const Hello({required this.clientId, this.gatewayId});
}

final class Event extends RelayServerToClientFrame {
  final RelayTerminalEvent event;
  const Event({required this.event});
}

final class ReplayDone extends RelayServerToClientFrame {
  final String sessionId;
  final int latestEventId;
  const ReplayDone({required this.sessionId, required this.latestEventId});
}

final class TetherError extends RelayServerToClientFrame {
  final String code;
  final String message;
  final String? sessionId;
  const TetherError({required this.code, required this.message, this.sessionId});
}

import 'package:flutter_test/flutter_test.dart';
import 'package:tether/models/protocol.dart';

void main() {
  group('RelayClientToServerFrame', () {
    test('ClientAuth.toJson includes token', () {
      final f = ClientAuth(token: 'tok');
      expect(f.toJson(), {'type': 'client.auth', 'token': 'tok'});
    });
    test('ClientList.toJson', () {
      expect(const ClientList().toJson(), {'type': 'client.list'});
    });
    test('ClientSubscribe.toJson includes mode and sessionId', () {
      final f = ClientSubscribe(sessionId: 's1', mode: RelayClientMode.control, after: 10);
      final j = f.toJson();
      expect(j['type'], 'client.subscribe');
      expect(j['sessionId'], 's1');
      expect(j['mode'], 'control');
      expect(j['after'], 10);
    });
    test('ClientStop.toJson', () {
      expect(ClientStop(sessionId: 'sid').toJson(), {'type': 'client.stop', 'sessionId': 'sid'});
    });
  });

  group('RelayServerToClientFrame.fromJson', () {
    test('client.auth.ok', () {
      final f = RelayServerToClientFrame.fromJson({'type': 'client.auth.ok', 'clientId': 'abc'});
      expect(f, isA<ClientAuthOk>());
      expect((f as ClientAuthOk).clientId, 'abc');
    });
    test('sessions parses list', () {
      final f = RelayServerToClientFrame.fromJson({
        'type': 'sessions',
        'sessions': [
          {
            'id': 's1', 'provider': 'claude', 'title': 'Test',
            'projectPath': '/tmp', 'status': 'running',
            'transport': 'pty-event-stream', 'lastActiveAt': 1000,
          }
        ],
      });
      expect(f, isA<Sessions>());
      expect((f as Sessions).sessions.first.id, 's1');
    });
    test('replay.done', () {
      final f = RelayServerToClientFrame.fromJson({
        'type': 'replay.done', 'sessionId': 's1', 'latestEventId': 42,
      });
      expect(f, isA<ReplayDone>());
      expect((f as ReplayDone).latestEventId, 42);
    });
    test('unknown type throws FormatException', () {
      expect(
        () => RelayServerToClientFrame.fromJson({'type': 'unknown.frame'}),
        throwsA(isA<FormatException>()),
      );
    });
  });

  group('RelaySession.fromJson', () {
    test('maps required fields', () {
      final s = RelaySession.fromJson({
        'id': 'sid', 'provider': 'codex', 'title': 'My Session',
        'projectPath': '/home/user', 'status': 'stopped',

      expect(s.status, RelaySessionStatus.stopped);
      expect(s.lastActiveAt, 9999);
    });
  });

    test('running serialization round-trip', () {
      expect(RelaySessionStatus.fromString('running'), RelaySessionStatus.running);
      expect(RelaySessionStatus.running.toJson(), 'running');
    });
  });
}

class ThemeNotifier extends ChangeNotifier {
  static const _key = 'tether:theme';
  ThemeMode _mode = ThemeMode.system;
  ThemeMode get mode => _mode;
  ThemeData get dark => _buildDarkTheme();
  ThemeData get light => _buildLightTheme();

  Future<void> load() async {
    final prefs = await SharedPreferences.getInstance();
    final stored = prefs.getString(_key);
    _mode = stored == 'dark' ? ThemeMode.dark
        : stored == 'light' ? ThemeMode.light
        : ThemeMode.system;
    notifyListeners();
  }

  Future<void> toggle() async {
    // 循环：system → dark → light → system
    _mode = switch (_mode) {
      ThemeMode.system => ThemeMode.dark,
      ThemeMode.dark   => ThemeMode.light,
      ThemeMode.light  => ThemeMode.system,
    };
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString(_key, _mode.name);
    notifyListeners();
  }
}

class LocaleNotifier extends ChangeNotifier {
  static const _key = 'tether:locale';
  Locale _locale = const Locale('zh');
  Locale get locale => _locale;

  Future<void> load() async {
    final prefs = await SharedPreferences.getInstance();
    final stored = prefs.getString(_key);
    _locale = stored == 'en' ? const Locale('en') : const Locale('zh');
    notifyListeners();
  }

  Future<void> toggle() async {
    _locale = _locale.languageCode == 'zh' ? const Locale('en') : const Locale('zh');
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString(_key, _locale.languageCode);
    notifyListeners();
  }
}
