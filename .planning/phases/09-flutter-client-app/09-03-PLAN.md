// Chat-first 数据模型 — 镜像 apps/web/src/components/session/chat-session-surface.tsx
// ConversationTurn sealed class 供 ChatSessionSurface 渲染

enum ConversationTurnStatus {
  complete,
  thinking, // 正在生成
  error,
}

// 工具调用信息
final class ToolCallInfo {
  final String toolCallId;
  final String toolName;
  final Map<String, dynamic> input;

  const ToolCallInfo({
    required this.toolCallId,
    required this.toolName,
    required this.input,
  });
}

// 选项信息（select.options 事件）
final class SelectOption {
  final String id;
  final String label;
  const SelectOption({required this.id, required this.label});

  factory SelectOption.fromJson(Map<String, dynamic> json) => SelectOption(
    id:    json['id'] as String,
    label: json['label'] as String,
  );
}

// ConversationTurn — 对话轮次 sealed class
// 镜像 Web ChatTurn 结构（chat-session-surface.tsx 数据模型）
sealed class ConversationTurn {
  final String id; // 唯一 id，用于 widget key
  const ConversationTurn({required this.id});
}

// 用户发送的消息
final class UserTurn extends ConversationTurn {
  final String content;
  const UserTurn({required super.id, required this.content});
}

// Assistant 生成的文字轮次（支持 markdown）
final class AssistantTurn extends ConversationTurn {
  final String content;
  final ConversationTurnStatus status;
  const AssistantTurn({
    required super.id,
    required this.content,
    this.status = ConversationTurnStatus.complete,
  });

  AssistantTurn copyWith({String? content, ConversationTurnStatus? status}) =>
      AssistantTurn(
        id:      id,
        content: content ?? this.content,
        status:  status  ?? this.status,
      );
}

// 工具调用 card
final class ToolCallTurn extends ConversationTurn {
  final ToolCallInfo toolCall;
  const ToolCallTurn({required super.id, required this.toolCall});
}

// 工具结果 card
final class ToolResultTurn extends ConversationTurn {
  final String toolCallId;
  final String output;
  const ToolResultTurn({
    required super.id,
    required this.toolCallId,
    required this.output,
  });
}

// 选项列表（select prompts）
final class SelectOptionsTurn extends ConversationTurn {
  final List<SelectOption> options;
  const SelectOptionsTurn({required super.id, required this.options});
}

// lib/services/conversation_service.dart
// Chat-first 数据服务 — 替代 LanClient（D-25 Relay only，LAN 移除）
// 职责：
//   1. 发送 client.chat 帧（通过 RelayClient.sendChat）
//   2. 监听 RelayClient.eventStream，将 agent 事件转换为 ConversationTurn
//   3. 暴露 turns 列表 + isTyping 状态供 ChatSessionSurface 渲染
// 参考：apps/web/src/components/session/chat-session-surface.tsx 数据模型
import 'dart:async';
import 'package:flutter/foundation.dart';
import '../models/conversation.dart';
import '../models/protocol.dart';
import 'relay_client.dart';

class ConversationService extends ChangeNotifier {
  // 可观察状态
  List<ConversationTurn> turns = [];
  bool isTyping = false;

  // 内部状态
  RelayClient? _relayClient;
  String? _sessionId;
  StreamSubscription<Event>? _eventSub;
  int _turnCounter = 0;

  String _nextId() => 'turn-${_turnCounter++}';

  // 绑定 RelayClient 和 sessionId（由 SessionScreen 在 initState 中调用）
  void attach(RelayClient relayClient, String sessionId) {
    // 解绑旧订阅
    _eventSub?.cancel();
    _relayClient = relayClient;
    _sessionId   = sessionId;
    turns        = [];
    isTyping     = false;
    _turnCounter = 0;

    // 监听 event stream，过滤当前 sessionId 的事件
    _eventSub = relayClient.eventStream.listen((eventFrame) {
      if (eventFrame.event.sessionId != sessionId) return;
      _handleEvent(eventFrame.event);
    });
  }

  // 发送用户消息：乐观添加 UserTurn，同时通过 RelayClient.sendChat 发送帧
  void sendMessage(String text) {
    if (_relayClient == null || _sessionId == null || text.trim().isEmpty) return;

    // 乐观 UI：立即添加 UserTurn
    turns = [...turns, UserTurn(id: _nextId(), content: text.trim())];
    notifyListeners();

    // 发送 client.chat 帧
    _relayClient!.sendChat(_sessionId!, text.trim());
  }

  // 处理来自 eventStream 的 RelayTerminalEvent
  // event.type 值：agent.turn / agent.thinking / chat.message / tool.call / tool.result / select.options
  void _handleEvent(RelayTerminalEvent event) {
    switch (event.type) {
      case 'agent.turn':
        // Assistant 生成的文字内容（可能分段到达，追加到最后一个 AssistantTurn）
        final content = event.payload['content'] as String? ?? '';
        if (turns.isNotEmpty && turns.last is AssistantTurn) {
          final last = turns.last as AssistantTurn;
          turns = [...turns.sublist(0, turns.length - 1),
            last.copyWith(
              content: last.content + content,
              status: ConversationTurnStatus.complete,
            )];
        } else {
          turns = [...turns, AssistantTurn(
            id: _nextId(),
            content: content,
            status: ConversationTurnStatus.complete,
          )];
        }
        isTyping = false;
        notifyListeners();

      case 'agent.thinking':
        // 思考中状态：添加或更新 AssistantTurn(status: thinking)
        final text = event.payload['text'] as String? ?? '';
        if (turns.isNotEmpty && turns.last is AssistantTurn) {
          final last = turns.last as AssistantTurn;
          turns = [...turns.sublist(0, turns.length - 1),
            last.copyWith(status: ConversationTurnStatus.thinking)];
        } else if (text.isNotEmpty) {
          turns = [...turns, AssistantTurn(
            id: _nextId(),
            content: text,
            status: ConversationTurnStatus.thinking,
          )];
        }
        isTyping = true;
        notifyListeners();

      case 'chat.message':
        // chat.message 确认（role=user 时忽略，因为 UserTurn 已乐观添加）
        final role = event.payload['role'] as String? ?? '';
        if (role == 'assistant') {
          final content = event.payload['content'] as String? ?? '';
          turns = [...turns, AssistantTurn(id: _nextId(), content: content)];
          isTyping = false;
          notifyListeners();
        }

      case 'tool.call':
        final toolCallId = event.payload['toolCallId'] as String? ?? _nextId();
        final toolName   = event.payload['toolName'] as String? ?? 'unknown';
        final input      = Map<String, dynamic>.from(
            event.payload['input'] as Map? ?? {});
        turns = [...turns, ToolCallTurn(
          id: _nextId(),
          toolCall: ToolCallInfo(
            toolCallId: toolCallId,
            toolName:   toolName,
            input:      input,
          ),
        )];
        notifyListeners();

      case 'tool.result':
        final toolCallId = event.payload['toolCallId'] as String? ?? '';
        final output     = event.payload['output'] as String? ?? '';
        turns = [...turns, ToolResultTurn(
          id:         _nextId(),
          toolCallId: toolCallId,
          output:     output,
        )];
        notifyListeners();

      case 'select.options':
        final rawOptions = event.payload['options'] as List? ?? [];
        final options = rawOptions
            .map((o) => SelectOption.fromJson(o as Map<String, dynamic>))
            .toList();
        if (options.isNotEmpty) {
          turns = [...turns, SelectOptionsTurn(id: _nextId(), options: options)];
          notifyListeners();
        }

      default:
        // 其他 event type（terminal.output 等）忽略
        break;
    }
  }

  // 清空对话历史（切换 session 时调用）
  void clear() {
    turns    = [];
    isTyping = false;
    _turnCounter = 0;
    notifyListeners();
  }

  @override
  void dispose() {
    _eventSub?.cancel();
    super.dispose();
  }
}

import 'dart:async';
import 'package:flutter_test/flutter_test.dart';
import 'package:tether/models/conversation.dart';
import 'package:tether/models/protocol.dart';
import 'package:tether/services/conversation_service.dart';

// 伪造 RelayClient：只实现 eventStream 和 sendChat 记录
class _FakeRelayClient extends RelayClient {
  final StreamController<Event> _ctrl = StreamController<Event>.broadcast();
  final List<({String sessionId, String message})> sentChats = [];

  @override
  Stream<Event> get eventStream => _ctrl.stream;

  @override
  void sendChat(String sessionId, String message) {
    sentChats.add((sessionId: sessionId, message: message));
  }

  void addEvent(RelayTerminalEvent e) => _ctrl.add(Event(event: e));

  @override
  void dispose() {
    _ctrl.close();
    super.dispose();
  }
}

RelayTerminalEvent makeEvent(String sessionId, String type, Map<String, dynamic> payload) =>
    RelayTerminalEvent(id: 0, sessionId: sessionId, type: type, ts: 0, payload: payload);

void main() {

    late _FakeRelayClient relay;
    late ConversationService svc;

    setUp(() {
      relay = _FakeRelayClient();
      svc   = ConversationService();

    tearDown(() {
      svc.dispose();
      relay.dispose();
    });

  });
