```dart
// lib/services/lan_client.dart
// LAN 直连服务 — 镜像 apps/web/src/main.tsx refreshDirect (lines 405-450)
// 两个功能:
//   1. 定期轮询 GET /api/sessions（与 Relay 模式的 client.list 等价，3s 周期）
//   2. WS ticket 流: POST /api/ws-ticket → open WS（不在 URL 暴露 access token）
// 安全: access token 只在 Authorization header 传递，不放 WS URL query param
import 'dart:async';
import 'dart:convert';
import 'dart:io';
import 'package:dio/dio.dart';
import 'package:flutter/foundation.dart';
import '../models/protocol.dart';
import 'relay_client.dart' show splitSessions; // 复用 splitSessions 函数

class LanClient extends ChangeNotifier {
  // 可观察状态（与 RelayClient 接口对齐）
  List<RelaySession> activeSessions = [];
  List<RelaySession> historySessions = [];
  bool hasLoaded = false;
  String status = '';

  // 内部
  final Dio _dio = Dio();
  Timer? _pollTimer;
  WebSocket? _ws;
  bool _disposed = false;

  String? _gatewayUrl;
  String? _accessToken;

  // 构建 WS URL — 安全: ticket 在 query param，access token 不在 URL
  // 镜像 RESEARCH.md Pattern 5
  static String buildWsUrl({
    required String gatewayUrl,
    required String sessionId,
    required String ticket,
    required String mode,
  }) {
    final base = gatewayUrl.trim().replaceAll(RegExp(r'/$'), '');
    final uri = Uri.parse(base);
    // http → ws; https → wss
    final wsScheme = uri.scheme == 'https' ? 'wss' : 'ws';
    final wsUri = Uri(
      scheme: wsScheme,
      host: uri.host,
      port: uri.port,
      path: '/api/sessions/${Uri.encodeComponent(sessionId)}/stream',
      queryParameters: {
        'ticket': ticket,
        'mode': mode,
        'surface': 'flutter',
      },
    );
    return wsUri.toString();
  }

  // 启动 LAN 模式：开始轮询（镜像 web setInterval(refreshDirect, 3000)）
  void start(String gatewayUrl, String accessToken) {
    _gatewayUrl = gatewayUrl;
    _accessToken = accessToken;
    _dio.options = BaseOptions(
      baseUrl: gatewayUrl,
      connectTimeout: const Duration(seconds: 5),
      receiveTimeout: const Duration(seconds: 5),
    );
    hasLoaded = false;
    notifyListeners();
    _fetchSessions();
    _pollTimer?.cancel();
    _pollTimer = Timer.periodic(const Duration(seconds: 3), (_) => _fetchSessions());
  }

  void stop() {
    _pollTimer?.cancel();
    _pollTimer = null;
    _ws?.close();
    _ws = null;
  }

  // 获取 session 列表（镜像 refreshDirect）
  Future<void> _fetchSessions() async {
    if (_disposed || _gatewayUrl == null || _accessToken == null) return;
    try {
      // GET /api/sessions（活跃）
      final res = await _dio.get(
        '/api/sessions',
        options: Options(headers: {'Authorization': 'Bearer $_accessToken'}),
      );
      final rawSessions = res.data['sessions'] as List? ?? [];
      final all = rawSessions
          .map((e) => RelaySession.fromJson(e as Map<String, dynamic>))
          .toList();

      // GET /api/sessions?all=1（历史）
      final histRes = await _dio.get(
        '/api/sessions',
        queryParameters: {'all': '1'},
        options: Options(headers: {'Authorization': 'Bearer $_accessToken'}),
      );
      final rawAll = histRes.data['sessions'] as List? ?? [];
      final allSessions = rawAll
          .map((e) => RelaySession.fromJson(e as Map<String, dynamic>))
          .toList();

      final split = splitSessions(allSessions);
      activeSessions = split.active;
      historySessions = split.history;
      hasLoaded = true;
      status = 'ok';
      notifyListeners();
    } on DioException catch (e) {
      if (e.response?.statusCode == 401) {
        status = 'auth_error';
        hasLoaded = true;
        notifyListeners();
      } else {
        status = 'unreachable';
        hasLoaded = true;
        notifyListeners();
      }
    } catch (_) {
      status = 'error';
      hasLoaded = true;
      notifyListeners();
    }
  }

  // 连接 WS ticket 流（订阅 PTY）
  // 安全: access token 只在 POST header 发送，ticket 用于 WS URL — per RESEARCH.md security
  Future<void> connectToSession({
    required String sessionId,
    required String mode,
    required void Function(Map<String, dynamic> frame) onFrame,
    required void Function() onDone,
  }) async {
    if (_gatewayUrl == null || _accessToken == null) return;

    // Step 1: 获取 WS ticket
    final ticketRes = await _dio.post(
      '/api/ws-ticket',
      data: {'sessionId': sessionId, 'mode': mode, 'surface': 'flutter'},
      options: Options(headers: {'Authorization': 'Bearer $_accessToken'}),
    );
    final ticket = ticketRes.data['ticket'] as String;

    // Step 2: 使用 ticket 打开 WS（不再需要 Authorization header）
    final wsUrl = buildWsUrl(
      gatewayUrl: _gatewayUrl!,
      sessionId: sessionId,
      ticket: ticket,
      mode: mode,
    );
    _ws?.close();
    _ws = await WebSocket.connect(wsUrl);

    _ws!.listen(
      (data) {
        if (data is String) {
          try {
            final frame = jsonDecode(data) as Map<String, dynamic>;
            onFrame(frame);
          } catch (_) {}
        }
      },
      onDone: () {
        if (!_disposed) onDone();
      },
      onError: (_) {
        if (!_disposed) onDone();
      },
      cancelOnError: false,
    );
  }

  void sendInput(String sessionId, String data) {
    _sendFrame({'type': 'client.input', 'sessionId': sessionId, 'data': data});
  }

  void sendResize(String sessionId, int cols, int rows) {
    _sendFrame({'type': 'client.resize', 'sessionId': sessionId, 'cols': cols, 'rows': rows});
  }

  // LAN 停止 session: POST /api/sessions/:id/stop（HTTP，不需要 WS）
  Future<void> stopSession(String sessionId) async {
    if (_gatewayUrl == null || _accessToken == null) return;
    try {
      await _dio.post(
        '/api/sessions/${Uri.encodeComponent(sessionId)}/stop',
        options: Options(headers: {'Authorization': 'Bearer $_accessToken'}),
      );
      // 乐观更新
      activeSessions = activeSessions.where((s) => s.id != sessionId).toList();
      notifyListeners();
      await _fetchSessions(); // 刷新列表
    } catch (_) {
      // 忽略停止失败，下次轮询会更新
    }
  }

  void _sendFrame(Map<String, dynamic> frame) {
    if (_ws?.readyState == WebSocket.open) {
      _ws!.add(jsonEncode(frame));
    }
  }

  @override
  void dispose() {
    _disposed = true;
    _pollTimer?.cancel();
    _ws?.close();
    super.dispose();
  }
}
```

```dart
import 'package:flutter_test/flutter_test.dart';
import 'package:tether/services/lan_client.dart';
import 'package:tether/models/protocol.dart';

void main() {
  group('LanClient.buildWsUrl', () {
    test('http → ws scheme', () {
      final url = LanClient.buildWsUrl(
        gatewayUrl: 'http://192.168.1.100:4789',
        sessionId: 's1',
        ticket: 'tok123',
        mode: 'control',
      );
      expect(url, startsWith('ws://'));
      expect(url, contains('192.168.1.100:4789'));
    });

    test('https → wss scheme', () {
      final url = LanClient.buildWsUrl(
        gatewayUrl: 'https://myhost.local:4789',
        sessionId: 's1',
        ticket: 'tok',
        mode: 'observe',
      );
      expect(url, startsWith('wss://'));
    });

    test('ticket included in query params', () {
      final url = LanClient.buildWsUrl(
        gatewayUrl: 'http://192.168.1.1:4789',
        sessionId: 's1',
        ticket: 'abc-ticket',
        mode: 'control',
      );
      expect(url, contains('ticket=abc-ticket'));
    });

    test('surface=flutter in query params', () {
      final url = LanClient.buildWsUrl(
        gatewayUrl: 'http://192.168.1.1:4789',
        sessionId: 's1',
        ticket: 'tok',
        mode: 'control',
      );
      expect(url, contains('surface=flutter'));
    });

    test('sessionId is URL-encoded in path', () {
      final url = LanClient.buildWsUrl(
        gatewayUrl: 'http://192.168.1.1:4789',
        sessionId: 'session with spaces',
        ticket: 'tok',
        mode: 'control',
      );
      expect(url, isNot(contains('session with spaces')));
      expect(url, contains('session%20with%20spaces'));
    });

    test('mode is included in query params', () {
      final url = LanClient.buildWsUrl(
        gatewayUrl: 'http://192.168.1.1:4789',
        sessionId: 's1',
        ticket: 'tok',
        mode: 'observe',
      );
      expect(url, contains('mode=observe'));
    });
  });

  group('LanClient initial state', () {
    test('activeSessions is empty', () {
      final client = LanClient();
      expect(client.activeSessions, isEmpty);
      client.dispose();
    });

    test('hasLoaded is false initially', () {
      final client = LanClient();
      expect(client.hasLoaded, false);
      client.dispose();
    });
  });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| App → Gateway HTTP | HTTP/HTTPS；Bearer token 在 Authorization header |
| App → Gateway WS | ws/wss；短生命周期 ticket 在 URL query（不含 access token）|

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-03-01 | Information Disclosure | LAN 直连 WS URL token 暴露 | mitigate | WS URL 只包含 ticket（短生命周期，单次使用）。Access token 只在 POST /api/ws-ticket 的 Authorization header 中传递，不出现在 WS URL。acceptance_criteria 中包含 grep 检查 |
| T-09-03-02 | Spoofing | LAN Gateway 证书验证 | accept | LAN 模式通常为 http://（内网）。如用户配置 https:// 则使用系统默认 TLS 验证（dart:io WebSocket 默认行为），不接受无效证书。不添加 acceptBadCertificate: true |
| T-09-03-03 | Denial of Service | 轮询 timer 3s 频率 | accept | 3s 轮询为内网直连场景合理频率，与 Web 端一致。dispose() 正确取消 timer 防止 timer 泄漏 |
| T-09-03-04 | Information Disclosure | 401 响应处理 | mitigate | 401 时设置 status='auth_error'，不暴露 token 内容到 UI 或日志 |
</threat_model>
