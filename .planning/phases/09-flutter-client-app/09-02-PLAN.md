// lib/services/auth_service.dart
// 鉴权服务 — 封装 login/register/token refresh/secure storage
// 镜像 apps/web/src/contexts/auth-context.tsx loginNormal/validateNormalSession/refreshNormal
// 关键安全约束：所有 token 只写 flutter_secure_storage，绝不使用 SharedPreferences
import 'package:dio/dio.dart';
import 'package:flutter/foundation.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';

// 默认 Server URL — per D-08，编译时常量
const String kDefaultServerUrl = 'https://tether.earntools.me';

// 默认 Relay URL — per D-09，login 响应未返回 relayUrl 时使用此回退
const String kDefaultRelayUrl = 'wss://tether.earntools.me';

class AuthException implements Exception {
  final String message;
  const AuthException(this.message);
  @override
  String toString() => 'AuthException: $message';
}

class AuthService extends ChangeNotifier {
  static const _storage    = FlutterSecureStorage();
  static const _accessKey  = 'tether:access_token';
  static const _refreshKey = 'tether:refresh_token';
  static const _relayUrlKey = 'tether:relay_url';

  // 主 Dio：附带 TokenRefreshInterceptor，处理 HTTP 请求 401 自动刷新
  final Dio _dio;
  // 独立刷新 Dio：无任何自定义 interceptor，供 _tryRefresh() 使用
  // 避免 /refresh 端点的 401 再次触发 QueuedInterceptor 递归（Codex Review Issue #7）
  final Dio _refreshDio;
  final String serverUrl;

  bool _isAuthenticated = false;
  bool get isAuthenticated => _isAuthenticated;

  // dio 和 refreshDio 参数允许测试注入 mock adapter
  AuthService({String? serverUrl, Dio? dio, Dio? refreshDio})
      : serverUrl = serverUrl ?? kDefaultServerUrl,
        _dio = dio ?? Dio(BaseOptions(
          baseUrl: serverUrl ?? kDefaultServerUrl,
          connectTimeout: const Duration(seconds: 10),
          receiveTimeout: const Duration(seconds: 10),
        )),
        _refreshDio = refreshDio ?? Dio(BaseOptions(
          baseUrl: serverUrl ?? kDefaultServerUrl,
          connectTimeout: const Duration(seconds: 10),
          receiveTimeout: const Duration(seconds: 10),
        )) {
    if (dio == null) {
      _dio.interceptors.add(
        TokenRefreshInterceptor(_storage, serverUrl ?? kDefaultServerUrl),
      );
    }
  }

  Future<String?> readAccessToken() => _storage.read(key: _accessKey);
  Future<String?> readRefreshToken() => _storage.read(key: _refreshKey);
  Future<String?> readRelayUrl()    => _storage.read(key: _relayUrlKey);

  // POST /login {email, password} — per D-07
  Future<void> login({required String email, required String password}) async {
    try {
      final res = await _dio.post('/login', data: {'email': email, 'password': password});
      final accessToken  = res.data['accessToken']  as String;
      final refreshToken = res.data['refreshToken'] as String;
      final relayUrl     = res.data['relayUrl'] as String?;
      await Future.wait([
        _storage.write(key: _accessKey,  value: accessToken),
        _storage.write(key: _refreshKey, value: refreshToken),
        if (relayUrl != null) _storage.write(key: _relayUrlKey, value: relayUrl),
      ]);
      _isAuthenticated = true;
      notifyListeners();
    } on DioException catch (e) {
      throw AuthException(e.response?.data?['message'] as String? ?? 'login failed');
    }
  }

  // POST /register {email, password, displayName?} — per D-07
  Future<void> register({
    required String email,
    required String password,
    String? displayName,
  }) async {
    try {
      final body = {
        'email': email,
        'password': password,
        if (displayName != null) 'displayName': displayName,
      };
      final res = await _dio.post('/register', data: body);
      final accessToken  = res.data['accessToken']  as String;
      final refreshToken = res.data['refreshToken'] as String;
      await Future.wait([
        _storage.write(key: _accessKey,  value: accessToken),
        _storage.write(key: _refreshKey, value: refreshToken),
      ]);
      _isAuthenticated = true;
      notifyListeners();
    } on DioException catch (e) {
      throw AuthException(e.response?.data?['message'] as String? ?? 'registration failed');
    }
  }

  // D-11: 启动时检查 token 有效性，失败则 silent refresh
  Future<bool> checkStoredToken() async {
    final access = await _storage.read(key: _accessKey);
    if (access == null) {
      _isAuthenticated = false;
      notifyListeners();
      return false;
    }
    try {
      await _dio.get(
        '/validate',
        options: Options(headers: {'Authorization': 'Bearer $access'}),
      );
      _isAuthenticated = true;
      notifyListeners();
      return true;
    } catch (_) {
      return _tryRefresh();
    }
  }

  // 使用 _refreshDio（独立，无拦截器），避免 401 递归
  Future<bool> _tryRefresh() async {
    final refresh = await _storage.read(key: _refreshKey);
    if (refresh == null) {
      await logout();
      return false;
    }
    try {
      final res = await _refreshDio.post(
        '/refresh',
        data: {'refreshToken': refresh},
      );
      final newAccess = res.data['accessToken'] as String;
      await _storage.write(key: _accessKey, value: newAccess);
      _isAuthenticated = true;
      notifyListeners();
      return true;
    } catch (_) {
      await logout();
      return false;
    }
  }

  Future<void> logout() async {
    await _storage.deleteAll();
    _isAuthenticated = false;
    notifyListeners();
  }
}

// QueuedInterceptor — 并发 401 只发起一次 refresh（不用于 /refresh 端点本身）
class TokenRefreshInterceptor extends QueuedInterceptor {
  final FlutterSecureStorage _store;
  final String _serverUrl;
  TokenRefreshInterceptor(this._store, this._serverUrl);

  @override
  Future<void> onError(DioException err, ErrorInterceptorHandler handler) async {
    if (err.response?.statusCode == 401) {
      final refresh = await _store.read(key: 'tether:refresh_token');
      if (refresh == null) {
        handler.next(err);
        return;
      }
      try {
        // 独立 Dio（无该拦截器），避免递归
        final res = await Dio(BaseOptions(baseUrl: _serverUrl))
            .post('/refresh', data: {'refreshToken': refresh});
        final newAccess = res.data['accessToken'] as String;
        await _store.write(key: 'tether:access_token', value: newAccess);
        handler.resolve(await Dio(BaseOptions(baseUrl: _serverUrl)).fetch(
          err.requestOptions
            ..headers['Authorization'] = 'Bearer $newAccess',
        ));
      } catch (_) {
        await _store.deleteAll();
        handler.next(err);
      }
    } else {
      handler.next(err);
    }
  }
}

// lib/services/relay_client.dart
// Relay WebSocket 客户端 — 镜像 apps/web/src/main.tsx lines 452-556 的状态机
// 8+8 协议：含 client.chat 发送 + replay.output 接收
// 退避: [1, 2, 4, 8, 16, 30] 秒上限 30s
// latestEventId: 存于服务层（非 widget）
import 'dart:async';
import 'dart:convert';
import 'dart:io';
import 'package:flutter/foundation.dart';
import '../models/protocol.dart';

// splitSessions — 镜像 apps/web/src/main.tsx splitActiveSessions
({List<RelaySession> active, List<RelaySession> history}) splitSessions(
    List<RelaySession> all) {
  final active = all.where((s) => s.status == RelaySessionStatus.running).toList();
  final activeIds = {for (final s in active) s.id};
  final history = all.where((s) => !activeIds.contains(s.id)).take(8).toList();
  return (active: active, history: history);
}

const _backoffSeconds = [1, 2, 4, 8, 16, 30];

int _computeBackoff(int index) =>
    _backoffSeconds[index.clamp(0, _backoffSeconds.length - 1)];

class RelayClient extends ChangeNotifier {
  // 可观察状态
  List<RelaySession> activeSessions  = [];
  List<RelaySession> historySessions = [];
  bool gatewayUnavailable = false;
  String status = '';
  bool hasLoaded = false;

  // 内部 WS 状态
  WebSocket? _ws;
  Timer? _listTimer;
  Timer? _reconnectTimer;
  int _backoffIndex = 0;
  bool _disposed = false;
  bool _authFailed = false; // 认证失败时停止重连
  String? _relayUrl;
  String? _accessToken;

  // 当前已订阅的 session（重连后重订阅用）
  String? _subscribedSessionId;
  RelayClientMode? _subscribedMode;

  // latestEventId 游标 — 存于服务层，不在 widget（per D-24）
  final Map<String, int> _latestEventIds = {};

  // Event stream — TerminalScreen 消费 PTY event 帧
  final StreamController<Event> _eventController =
      StreamController<Event>.broadcast();
  Stream<Event> get eventStream => _eventController.stream;

  // ReplayOutput stream — ReplayScreen 消费回放输出帧（Codex Review Issue #2）
  final StreamController<ReplayOutput> _replayOutputController =
      StreamController<ReplayOutput>.broadcast();
  Stream<ReplayOutput> get replayOutputStream => _replayOutputController.stream;

  int? getLatestEventId(String sessionId) => _latestEventIds[sessionId];

  void _updateLatestEventId(String sessionId, int eventId) {
    if ((_latestEventIds[sessionId] ?? -1) < eventId) {
      _latestEventIds[sessionId] = eventId;
    }
  }

  // Codex Review Issue #8: _buildRelayUri 不无条件 append /client
  // 逻辑：若 relayUrl 路径已以 /client 结尾 → 直接使用；否则 append /client
  String _buildRelayUri(String relayUrl) {
    final uri = Uri.parse(relayUrl.trim());
    // http/https → ws/wss
    final wsScheme = (uri.scheme == 'https' || uri.scheme == 'wss') ? 'wss' : 'ws';
    final currentPath = uri.path.replaceAll(RegExp(r'/$'), '');
    // 检测是否已含 /client 后缀（避免 wss://host/client/client 双重 append）
    final path = currentPath.endsWith('/client')
        ? currentPath
        : '$currentPath/client';
    return Uri(
      scheme: wsScheme,
      host: uri.host,
      port: uri.port != 0 ? uri.port : null,
      path: path,
    ).toString();
  }

  Future<void> connect(String relayUrl, String accessToken) async {
    _relayUrl    = relayUrl;
    _accessToken = accessToken;
    _authFailed  = false;
    _reconnectTimer?.cancel();
    _ws?.close();
    _ws = null;

    _updateStatus('connecting');
    hasLoaded = false;
    notifyListeners();

    try {
      final uri = _buildRelayUri(relayUrl);
      _ws = await WebSocket.connect(uri);
    } catch (_) {
      _scheduleReconnect();
      return;
    }

    _sendFrame(ClientAuth(token: accessToken));
    _updateStatus('auth');

    _ws!.listen(
      _handleMessage,
      onDone: _onClose,
      onError: (_) => _onClose(),
      cancelOnError: false,
    );
  }

  void _handleMessage(dynamic data) {
    if (_disposed) return;
    if (data is! String) return;
    Map<String, dynamic> raw;
    try {
      raw = jsonDecode(data) as Map<String, dynamic>;
    } catch (_) {
      return;
    }

    final RelayServerToClientFrame frame;
    try {
      frame = RelayServerToClientFrame.fromJson(raw);
    } on FormatException {
      return; // 未知帧忽略
    }

    switch (frame) {
      case ClientAuthOk(:final clientId):
        gatewayUnavailable = false;
        _authFailed = false;
        _backoffIndex = 0; // 重置退避
        _updateStatus('relay:${clientId.substring(0, clientId.length.clamp(0, 8))}');
        _sendFrame(const ClientList());
        _listTimer?.cancel();
        _listTimer = Timer.periodic(const Duration(seconds: 3), (_) {
          if (_ws?.readyState == WebSocket.open) {
            _sendFrame(const ClientList());
          }
        });
        // 重连后重新订阅（Codex Review P02 MEDIUM #3）
        if (_subscribedSessionId != null && _subscribedMode != null) {
          final after = _latestEventIds[_subscribedSessionId!];
          _sendFrame(ClientSubscribe(
            sessionId: _subscribedSessionId!,
            mode: _subscribedMode!,
            after: after,
          ));
        }

      case ClientAuthFailed(:final message):
        // 认证失败：停止重连（Codex Review Issue #2 / P02 MEDIUM）
        _authFailed = true;
        _updateStatus(message);
        hasLoaded = true;
        _ws?.close(); // close 触发 _onClose，但 _authFailed=true 阻止重连
        notifyListeners();

      case Sessions(:final sessions):
        final split = splitSessions(sessions);
        activeSessions  = split.active;
        historySessions = split.history;
        gatewayUnavailable = false;
        hasLoaded = true;
        notifyListeners();

      case Event(:final event):
        _updateLatestEventId(event.sessionId, event.id);
        _eventController.add(Event(event: event));
        notifyListeners();

      case ReplayOutput() as ro:
        // 新增：relay.output 帧广播到 replayOutputStream（Codex Review Issue #2）
        _updateLatestEventId(ro.sessionId, ro.latestEventId);
        _replayOutputController.add(ro);

      case ReplayDone(:final sessionId, :final latestEventId):
        _updateLatestEventId(sessionId, latestEventId);
        notifyListeners();

      case TetherError(:final code):
        if (code == 'gateway_unavailable') {
          activeSessions  = [];
          historySessions = [];
          gatewayUnavailable = true;
          hasLoaded = true;
          notifyListeners();
        } else {
          _updateStatus(code);
          hasLoaded = true;
          notifyListeners();
        }

      case Hello():
        break;
    }
  }

  void _onClose() {
    if (_disposed) return;
    _listTimer?.cancel();
    _listTimer = null;
    _updateStatus('closed');
    hasLoaded = true;
    notifyListeners();
    // 认证失败时不重连（Codex Review Issue #2）
    if (!_authFailed) {
      _scheduleReconnect();
    }
  }

  void _scheduleReconnect() {
    if (_disposed || _authFailed) return;
    final delay = _computeBackoff(_backoffIndex);
    if (_backoffIndex < _backoffSeconds.length - 1) _backoffIndex++;
    _reconnectTimer = Timer(Duration(seconds: delay), () {
      if (!_disposed && !_authFailed && _relayUrl != null && _accessToken != null) {
        connect(_relayUrl!, _accessToken!);
      }
    });
  }

  // D-23: 从后台回来立即重连
  void reconnectIfNeeded() {
    if (_ws == null || _ws!.readyState != WebSocket.open) {
      if (_relayUrl != null && _accessToken != null && !_authFailed) {
        _reconnectTimer?.cancel();
        connect(_relayUrl!, _accessToken!);
      }
    }
  }

  // 订阅 session（D-24: 使用 latestEventId 作为 after）
  void subscribe(String sessionId, RelayClientMode mode) {
    _subscribedSessionId = sessionId;
    _subscribedMode      = mode;
    final after = _latestEventIds[sessionId];
    _sendFrame(ClientSubscribe(sessionId: sessionId, mode: mode, after: after));
  }

  // Replay 订阅（D-37: mode=observe, after=0）
  void subscribeReplay(String sessionId) {
    _sendFrame(ClientSubscribe(
      sessionId: sessionId,
      mode: RelayClientMode.observe,
      after: 0,
    ));
  }

  // chat 能力（D-04a / Codex Review Issue #2）
  void sendChat(String sessionId, String message) {
    _sendFrame(ClientChat(sessionId: sessionId, message: message));
  }

  void sendInput(String sessionId, String data) {
    _sendFrame(ClientInput(sessionId: sessionId, data: data));
  }

  void sendResize(String sessionId, int cols, int rows) {
    _sendFrame(ClientResize(sessionId: sessionId, cols: cols, rows: rows));
  }

  void stopSession(String sessionId) {
    _sendFrame(ClientSubscribe(sessionId: sessionId, mode: RelayClientMode.control));
    _sendFrame(ClientStop(sessionId: sessionId));
    activeSessions = activeSessions.where((s) => s.id != sessionId).toList();
    notifyListeners();
  }

  void detach(String sessionId) {
    _sendFrame(ClientDetach(sessionId: sessionId));
  }

  void _sendFrame(RelayClientToServerFrame frame) {
    if (_ws?.readyState == WebSocket.open) {
      _ws!.add(jsonEncode(frame.toJson()));
    }
  }

  void _updateStatus(String s) {
    status = s;
    notifyListeners();
  }

  // 测试辅助（@visibleForTesting）
  @visibleForTesting
  void updateLatestEventIdForTest(String sessionId, int eventId) =>
      _updateLatestEventId(sessionId, eventId);

  // 测试辅助：暴露 _buildRelayUri
  @visibleForTesting
  String buildRelayUriForTest(String url) => _buildRelayUri(url);

  // 测试辅助：暴露 _computeBackoff（模块级函数包装）
  @visibleForTesting
  int computeBackoffForTest(int index) => _computeBackoff(index);

  @override
  void dispose() {
    _disposed = true;
    _listTimer?.cancel();
    _reconnectTimer?.cancel();
    _ws?.close();
    _eventController.close();
    _replayOutputController.close();
    super.dispose();
  }
}
