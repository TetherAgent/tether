```typescript
test('Phase17-T1: relay broadcasts agent.delta to all chat subscribers (same account)', async () => {
  const syncServer = createServer((request, response) => {
    let body = '';
    request.on('data', (chunk) => { body += chunk.toString(); });
    request.on('end', () => {
      const metadataMatch = request.url?.match(/^\/api\/relay\/gateway-sessions\/([^/]+)\/metadata$/);
      if (request.method === 'GET' && metadataMatch) {
        const sessionId = decodeURIComponent(metadataMatch[1]!);
        response.writeHead(200, { 'content-type': 'application/json' });
        response.end(JSON.stringify({
          code: 0, msg: 'success',
          data: { id: sessionId, provider: 'codex', projectPath: process.cwd(),
                  accountId: 'acct_a', gatewayId: 'gateway-a', userId: 'user_a', transport: 'chat' }
        }));
        return;
      }
      response.writeHead(200, { 'content-type': 'application/json' });
      response.end(JSON.stringify({ code: 0, msg: 'success', data: { events: [] } }));
    });
  });
  await new Promise<void>((resolve) => syncServer.listen(0, '127.0.0.1', resolve));
  const syncAddr = syncServer.address();
  assert(syncAddr && typeof syncAddr !== 'string');

  const relay = await startRelayServer({
    host: '127.0.0.1', port: 0, secret: SECRET,
    serverSyncUrl: `http://127.0.0.1:${syncAddr.port}`,
    runtimeSyncSecret: 'runtime-secret',
    validateToken: async (token) => {
      if (token === 'gw-a') return { accountId: 'acct_a', gatewayId: 'gateway-a', userId: 'user_a', tokenClass: 'gateway_access', expiresAt: Date.now() + 60_000, jti: 'gw_a' };
      if (token === 'client-a1') return { accountId: 'acct_a', gatewayId: 'gateway-a', userId: 'user_a', tokenClass: 'normal_client_access', expiresAt: Date.now() + 60_000, jti: 'cli_a1' };
      if (token === 'client-a2') return { accountId: 'acct_a', gatewayId: 'gateway-a', userId: 'user_a', tokenClass: 'normal_client_access', expiresAt: Date.now() + 60_000, jti: 'cli_a2' };
      return undefined;
    }
  });

  const gateway = new WebSocket(`${relay.url.replace('http', 'ws')}/ws/gateway`);
  const clientA1 = new WebSocket(`${relay.url.replace('http', 'ws')}/ws/client`);
  const clientA2 = new WebSocket(`${relay.url.replace('http', 'ws')}/ws/client`);

  try {
    await waitForOpen(gateway);
    gateway.send(JSON.stringify({ type: 'gateway.auth', gatewayId: 'gateway-a', token: 'gw-a' }));
    await waitForJson(gateway, (m) => (m as RelayServerToGatewayFrame).type === 'gateway.auth.ok');

    await waitForOpen(clientA1);
    clientA1.send(JSON.stringify({ type: 'client.auth', token: 'client-a1' }));
    await waitForJson(clientA1, (m) => (m as RelayServerToClientFrame).type === 'client.auth.ok');

    await waitForOpen(clientA2);
    clientA2.send(JSON.stringify({ type: 'client.auth', token: 'client-a2' }));
    await waitForJson(clientA2, (m) => (m as RelayServerToClientFrame).type === 'client.auth.ok');

    const SESSION_ID = 'tth_chat_multi_t1';
    clientA1.send(JSON.stringify({ type: 'client.subscribe', sessionId: SESSION_ID, mode: 'control', after: 0 }));
    await waitForJson(clientA1, (m) => (m as RelayServerToClientFrame).type === 'sessions');
    clientA2.send(JSON.stringify({ type: 'client.subscribe', sessionId: SESSION_ID, mode: 'control', after: 0 }));
    await waitForJson(clientA2, (m) => (m as RelayServerToClientFrame).type === 'sessions');

    // Gateway 发送 agent.delta
    gateway.send(JSON.stringify({
      type: 'gateway.event',
      gatewayId: 'gateway-a',
      event: { id: 101, type: 'agent.delta', sessionId: SESSION_ID, payload: { text: 'hello-multi', clientId: 'client-a1' } }
    }));

    const [delta1, delta2] = await Promise.all([
      waitForJson(clientA1, (m) => (m as RelayServerToClientFrame).type === 'agent.delta'),
      waitForJson(clientA2, (m) => (m as RelayServerToClientFrame).type === 'agent.delta')
    ]);
    assert.equal((delta1 as { sessionId: string }).sessionId, SESSION_ID);
    assert.equal((delta2 as { sessionId: string }).sessionId, SESSION_ID);
  } finally {
    gateway.close(); clientA1.close(); clientA2.close();
    await relay.close();
    await new Promise<void>((resolve) => syncServer.close(() => resolve()));
  }
});
```
