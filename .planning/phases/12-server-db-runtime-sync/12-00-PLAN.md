```typescript
 * Phase 12: Server runtime-sync 写接口测试

 * Wave 0 stub — 在 Wave 2（12-02-PLAN.md）实现 runtimeSyncRepository 之前先占位。

 * - MySQL 未启用（测试环境）时 upsert 方法直接 return，不写 DB。
 * - secret header 校验由 middleware 完成，这里只测 service 层幂等行为。
 */
import assert from 'assert'
import { Context } from 'egg'
import { app } from 'egg-mock/bootstrap'

describe('test/runtime-sync.test.ts', () => {
  let ctx: Context

  beforeEach(async () => {
    ctx = app.mockContext()
  })

  it('runtimeSyncRepository.upsertGatewaySession — MySQL 未启用时静默返回', async () => {
    // 测试环境 mysqlModeEnabled() 返回 false，upsert 应立即 return 不抛出
    const scope = { accountId: 'acct_1', workspaceId: 'ws_1', gatewayId: 'gw_1' }
    await assert.doesNotReject(async () => {
      await (ctx.service as unknown as Record<string, unknown & { upsertGatewaySession: (...args: unknown[]) => Promise<void> }>).runtimeSyncRepository?.upsertGatewaySession?.(
        { id: 'tth_sync_01', provider: 'codex', status: 'running' },
        scope
      )
    })
  })

  it('runtimeSyncRepository.upsertChatMessage — MySQL 未启用时静默返回', async () => {
    const scope = { accountId: 'acct_1', workspaceId: 'ws_1', gatewayId: 'gw_1' }
    await assert.doesNotReject(async () => {
      await (ctx.service as unknown as Record<string, unknown & { upsertChatMessage: (...args: unknown[]) => Promise<void> }>).runtimeSyncRepository?.upsertChatMessage?.(
        'tth_sync_01', 1, 'user', 'hello', null, scope
      )
    })
  })

  it('runtimeSyncRepository.upsertRuntimeEvent — 非白名单事件不写入', async () => {
    // 非白名单事件（如 custom.event）即使调用也应无副作用
    const scope = { accountId: 'acct_1', workspaceId: 'ws_1', gatewayId: 'gw_1' }

      await (ctx.service as unknown as Record<string, unknown & { upsertRuntimeEvent: (...args: unknown[]) => Promise<void> }>).runtimeSyncRepository?.upsertRuntimeEvent?.(
        'tth_sync_01', 1, 'custom.not_whitelisted', { data: 'x' }, scope
      )
    })
  })
})
```

```typescript
test('relay gateway.event sync failure does not block frame forwarding', async () => {
  // syncToServer 指向一个不存在的端口，验证 sync 失败时不阻塞 gateway.event 的正常转发
  const relay = await startRelayServer({
    host: '127.0.0.1',
    port: 0,
    secret: SECRET,
    serverSyncUrl: 'http://127.0.0.1:1', // 无效地址，sync 必然失败
    runtimeSyncSecret: 'test-sync-secret',
    validateToken: async (token) => {
      if (token === GATEWAY_TOKEN) {
        return {
          accountId: 'acct_1',
          workspaceId: 'ws_1',
          gatewayId: 'gateway-test',
          tokenClass: 'gateway_access',
          expiresAt: Date.now() + 60_000,
          jti: 'jti_sync_test'
        };
      }
      if (token === CLIENT_TOKEN) {
        return {
          accountId: 'acct_1',
          workspaceId: 'ws_1',
          userId: 'user_1',
          tokenClass: 'normal_client_access',
          expiresAt: Date.now() + 60_000,
          jti: 'jti_sync_client'
        };
      }
      return undefined;
    }
  });
  const gateway = new WebSocket(`${relay.url.replace('http', 'ws')}/gateway`);
  const client = new WebSocket(`${relay.url.replace('http', 'ws')}/client`);

  try {
    await authenticateGateway(gateway);
    await authenticateClient(client);

    // 先发 sessions 让 client 可以接收 event
    gateway.send(JSON.stringify({
      type: 'gateway.sessions',
      gatewayId: 'gateway-test',
      sessions: [{
        id: 'tth_sync_fail_test',
        provider: 'codex',
        title: 'Sync Fail Test',
        projectPath: process.cwd(),
        accountId: 'acct_1',
        workspaceId: 'ws_1',
        gatewayId: 'gateway-test',
        userId: 'user_1',
        status: 'running',
        transport: 'pty-event-stream',
        lastActiveAt: Date.now()
      }]
    }));
    await waitForJson(client, (message) => message.type === 'sessions');

    // 订阅 session 以接收 event
    client.send(JSON.stringify({ type: 'client.subscribe', sessionId: 'tth_sync_fail_test', after: 0, mode: 'control' }));
    await waitForJson(gateway, (message) => message.type === 'client.subscribe');

    // 发送 terminal.output event（白名单事件，syncToServer 会尝试 POST 到无效地址）
    // 验证：即使 syncToServer 失败，client 仍然收到 event（转发不被阻塞，D-04、D-05）
    gateway.send(JSON.stringify({
      type: 'gateway.event',
      gatewayId: 'gateway-test',
      event: {
        id: 1,
        sessionId: 'tth_sync_fail_test',
        type: 'terminal.output',
        ts: Date.now(),
        payload: { data: 'hello\r\n', encoding: 'utf8' }
      }
    }));

    // client 收到 event（转发正常）
    const receivedEvent = await waitForJson(client, (message) => message.type === 'event');
    assert.equal((receivedEvent.event as Record<string, unknown>)?.type, 'terminal.output');
  } finally {
    gateway.close();
    client.close();
    await relay.close();
  }
});
```

注意：追加在文件最末尾的 `});` 之后，不改动已有任何 test case 或辅助函数。
