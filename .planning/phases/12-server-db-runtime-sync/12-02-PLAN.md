```typescript
import { Service } from 'egg';

// 白名单：这些 event_type 写 gateway_runtime_events
const RUNTIME_EVENT_WHITELIST = new Set([
  'terminal.output',
  'terminal.input',
  'session.error',
  'session.exited',
  'agent.status'
]);

// 掩码值与 apps/gateway/src/mask.ts 保持一致
const MASK = '[REDACTED]';

// 正则规则完整复制自 apps/gateway/src/mask.ts PATTERNS 数组（D-09 等效对齐）
// Server 不能跨 app import gateway/src/mask.ts，因此在此复制全部 4 条规则
const MASK_PATTERNS: RegExp[] = [
  /sk-[A-Za-z0-9_-]{20,}/g,
  /(?:api[_-]?key|token|secret|password)\s*[:=]\s*["']?[^"'\s]+/gi,
  /ghp_[A-Za-z0-9_]{20,}/g,
  /github_pat_[A-Za-z0-9_]{40,}/g
];

// 对齐 maskSensitiveOutput 的掩码逻辑（写入 terminal.input/output 前调用，per D-09）
function maskPayload(payload: unknown): string {
  const str = typeof payload === 'string' ? payload : JSON.stringify(payload ?? {});
  return MASK_PATTERNS.reduce((text, pattern) => text.replace(pattern, MASK), str);
}

// 限制 payload_json 长度（MEDIUMTEXT 上限但代码层保护 64KB）
const MAX_PAYLOAD_BYTES = 65536;

function truncatePayload(s: string): string {
  if (s.length > MAX_PAYLOAD_BYTES) {
    return s.slice(0, MAX_PAYLOAD_BYTES) + '...[TRUNCATED]';
  }
  return s;
}

type RelayAuthScope = {
  accountId: string;
  workspaceId: string;
  gatewayId: string;
};

export default class RuntimeSyncRepositoryService extends Service {
  private mysqlModeEnabled() {
    return this.ctx.service.db.mysqlModeEnabled();
  }

  // 写 gateway_sessions（upsert，幂等，PRIMARY KEY id）
  public async upsertGatewaySession(
    session: {
      id: string;
      provider: string;
      title?: string;
      projectPath?: string;
      agentSessionId?: string;
      status: string;
      transport?: string;
      lastActiveAt?: number;
      userId?: string;
    },
    scope: RelayAuthScope
  ): Promise<void> {
    if (!this.mysqlModeEnabled()) return;
    const now = new Date();
    await this.ctx.service.db.query(
      `INSERT INTO gateway_sessions
         (id, account_id, workspace_id, gateway_id, user_id, provider, title, project_path,
          agent_session_id, status, transport, last_active_at)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
       ON DUPLICATE KEY UPDATE
         status = VALUES(status),
         title = VALUES(title),
         project_path = VALUES(project_path),
         last_active_at = VALUES(last_active_at),
         updated_at = CURRENT_TIMESTAMP`,
      [
        session.id,
        scope.accountId,
        scope.workspaceId,
        scope.gatewayId,
        session.userId ?? null,
        session.provider,
        session.title ?? null,
        session.projectPath ?? null,
        session.agentSessionId ?? null,
        session.status,
        session.transport ?? 'pty-event-stream',
        session.lastActiveAt ? new Date(session.lastActiveAt) : now
      ]
    );
  }

  // 写 gateway_chat_messages（upsert，幂等，UNIQUE session_id+turn_index）
  // 写前校验 session 归属（防串 Pitfall 3）
  public async upsertChatMessage(
    sessionId: string,
    turnIndex: number,
    role: string,
    content: string,
    toolsJson: string | null,
    scope: RelayAuthScope
  ): Promise<void> {
    if (!this.mysqlModeEnabled()) return;
    // 防串校验：session 必须属于当前 gateway/account/workspace
    const rows = await this.ctx.service.db.query(
      `SELECT id FROM gateway_sessions
       WHERE id = ? AND gateway_id = ? AND account_id = ? AND workspace_id = ? LIMIT 1`,
      [sessionId, scope.gatewayId, scope.accountId, scope.workspaceId]
    );
    if (!Array.isArray(rows) || rows.length === 0) {
      console.warn(`[server] upsertChatMessage: session ${sessionId} not found or scope mismatch`);
      return;
    }
    await this.ctx.service.db.query(
      `INSERT INTO gateway_chat_messages (session_id, turn_index, role, content, tools_json)
       VALUES (?, ?, ?, ?, ?)
       ON DUPLICATE KEY UPDATE
         role = VALUES(role),
         content = VALUES(content),
         tools_json = VALUES(tools_json),
         updated_at = CURRENT_TIMESTAMP`,
      [sessionId, turnIndex, role, content, toolsJson]
    );
  }

  // 写 gateway_runtime_events（upsert，幂等，UNIQUE session_id+event_id）
  // 只写白名单事件；terminal.input/output 先掩码（D-09）；写前校验归属
  public async upsertRuntimeEvent(
    sessionId: string,
    eventId: number,
    eventType: string,
    payload: unknown,
    scope: RelayAuthScope
  ): Promise<void> {
    if (!this.mysqlModeEnabled()) return;
    if (!RUNTIME_EVENT_WHITELIST.has(eventType)) return;
    // 防串校验
    const rows = await this.ctx.service.db.query(
      `SELECT id FROM gateway_sessions
       WHERE id = ? AND gateway_id = ? AND account_id = ? AND workspace_id = ? LIMIT 1`,
      [sessionId, scope.gatewayId, scope.accountId, scope.workspaceId]
    );
    if (!Array.isArray(rows) || rows.length === 0) {
      console.warn(`[server] upsertRuntimeEvent: session ${sessionId} scope mismatch`);
      return;
    }
    // 掩码 + 长度限制（D-09）
    // terminal.input / terminal.output 必须掩码，其他白名单事件也统一经过 maskPayload
    const rawJson = typeof payload === 'string' ? payload : JSON.stringify(payload ?? {});
    const payloadJson = truncatePayload(maskPayload(rawJson));
    await this.ctx.service.db.query(
      `INSERT INTO gateway_runtime_events (session_id, event_id, event_type, payload_json)
       VALUES (?, ?, ?, ?)
       ON DUPLICATE KEY UPDATE
         event_type = VALUES(event_type),
         payload_json = VALUES(payload_json),
         updated_at = CURRENT_TIMESTAMP`,
      [sessionId, eventId, eventType, payloadJson]
    );
  }

  // 写 gateway_sync_cursors（upsert，UNIQUE gateway_id+session_id）
  public async upsertSyncCursor(
    sessionId: string,
    lastEventId: number | null,
    lastTurnIndex: number | null,
    scope: RelayAuthScope
  ): Promise<void> {
    if (!this.mysqlModeEnabled()) return;
    await this.ctx.service.db.query(
      `INSERT INTO gateway_sync_cursors (gateway_id, session_id, last_event_id, last_turn_index, last_synced_at)
       VALUES (?, ?, ?, ?, CURRENT_TIMESTAMP)
       ON DUPLICATE KEY UPDATE
         last_event_id = VALUES(last_event_id),
         last_turn_index = VALUES(last_turn_index),
         last_synced_at = CURRENT_TIMESTAMP,
         updated_at = CURRENT_TIMESTAMP`,
      [scope.gatewayId, sessionId, lastEventId, lastTurnIndex]
    );
  }
}
```

1. 创建 apps/server/app/controller/runtime-sync.ts：
```typescript
import { Controller } from 'egg';

export default class RuntimeSyncController extends Controller {
  // POST /api/runtime-sync/gateway/sessions
  // body: { gatewayId, sessions: Array<{id, provider, title, projectPath, agentSessionId, status, transport, lastActiveAt, userId}>, scope: {accountId, workspaceId, gatewayId} }
  public async sessions(): Promise<void> {
    const { ctx } = this;
    const body = ctx.request.body as Record<string, unknown>;
    const sessions = (body.sessions as Array<Record<string, unknown>>) ?? [];
    const scope = body.scope as { accountId: string; workspaceId: string; gatewayId: string };
    if (!scope?.accountId || !scope?.workspaceId || !scope?.gatewayId) {
      ctx.throw(400, 'Missing scope');
    }
    for (const session of sessions) {
      await ctx.service.runtimeSyncRepository.upsertGatewaySession(
        {
          id: String(session.id ?? ''),
          provider: String(session.provider ?? ''),
          title: session.title ? String(session.title) : undefined,
          projectPath: session.projectPath ? String(session.projectPath) : undefined,
          agentSessionId: session.agentSessionId ? String(session.agentSessionId) : undefined,
          status: String(session.status ?? 'running'),
          transport: session.transport ? String(session.transport) : undefined,
          lastActiveAt: typeof session.lastActiveAt === 'number' ? session.lastActiveAt : undefined,
          userId: session.userId ? String(session.userId) : undefined
        },
        scope
      );
    }
    ctx.success({ ok: true });
  }

  // POST /api/runtime-sync/gateway/conversation
  // body: { sessionId, turns: Array<{turnIndex, role, content, tools}>, scope }
  public async conversation(): Promise<void> {
    const { ctx } = this;
    const body = ctx.request.body as Record<string, unknown>;
    const sessionId = String(body.sessionId ?? '');
    const turns = (body.turns as Array<Record<string, unknown>>) ?? [];
    const scope = body.scope as { accountId: string; workspaceId: string; gatewayId: string };
    if (!sessionId || !scope?.accountId || !scope?.workspaceId || !scope?.gatewayId) {
      ctx.throw(400, 'Missing sessionId or scope');
    }
    for (const turn of turns) {
      await ctx.service.runtimeSyncRepository.upsertChatMessage(
        sessionId,
        Number(turn.turnIndex ?? 0),
        String(turn.role ?? 'assistant'),
        String(turn.content ?? ''),
        turn.tools ? JSON.stringify(turn.tools) : null,
        scope
      );
    }
    // 更新同步 cursor
    const lastTurn = turns[turns.length - 1];
    if (lastTurn) {
      await ctx.service.runtimeSyncRepository.upsertSyncCursor(
        sessionId, null, Number(lastTurn.turnIndex ?? 0), scope
      );
    }
    ctx.success({ ok: true });
  }

  // POST /api/runtime-sync/gateway/event
  // body: { gatewayId, event: {id, sessionId, type, payload}, scope }
  // event.type === 'agent.turn' 写 gateway_chat_messages（通过 conversation 接口的 turn 逻辑）
  // event.type in WHITELIST 写 gateway_runtime_events
  public async event(): Promise<void> {
    const { ctx } = this;
    const body = ctx.request.body as Record<string, unknown>;
    const event = body.event as Record<string, unknown> | undefined;
    const scope = body.scope as { accountId: string; workspaceId: string; gatewayId: string };
    if (!event || !scope?.accountId || !scope?.workspaceId || !scope?.gatewayId) {
      ctx.throw(400, 'Missing event or scope');
    }
    const eventType = String(event.type ?? '');
    const sessionId = String(event.sessionId ?? '');
    const eventId = Number(event.id ?? 0);

    if (eventType === 'agent.turn') {
      // 写 gateway_chat_messages（agent.turn payload 含 role / content / tools / turnIndex）
      const payload = event.payload as Record<string, unknown> | undefined;
      await ctx.service.runtimeSyncRepository.upsertChatMessage(
        sessionId,
        Number(payload?.turnIndex ?? 0),
        String(payload?.role ?? 'assistant'),
        String(payload?.content ?? ''),
        payload?.tools ? JSON.stringify(payload.tools) : null,
        scope
      );
      await ctx.service.runtimeSyncRepository.upsertSyncCursor(
        sessionId, eventId, Number(payload?.turnIndex ?? 0), scope
      );
    } else {
      // 写 gateway_runtime_events（WHITELIST 过滤在 service 内部）
      await ctx.service.runtimeSyncRepository.upsertRuntimeEvent(
        sessionId, eventId, eventType, event.payload, scope
      );
      await ctx.service.runtimeSyncRepository.upsertSyncCursor(
        sessionId, eventId, null, scope
      );
```
