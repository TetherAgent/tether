正确：先 SELECT DISTINCT session_id，再对每个 session 单独执行：
DELETE FROM gateway_runtime_events
WHERE session_id = ?
  AND id < (

      SELECT id FROM gateway_runtime_events
      WHERE session_id = ?

  )

```typescript
import { Subscription } from 'egg';

// 每次批量删除超龄行的最大条数（避免长事务锁表）
const DELETE_BATCH_SIZE = 10000;
// 每个 session 保留的最新事件数量
const MAX_EVENTS_PER_SESSION = 100000;

export default class CleanupRuntimeEvents extends Subscription {
  static get schedule() {
    return {
      type: 'worker',
      cron: '0 0 3 * * *' // 每天凌晨 3 点执行
    };
  }

  async subscribe() {
    const { ctx } = this;
    if (!ctx.service.db.mysqlModeEnabled()) return;

    // 步骤 1：删除超过 1 个月的旧行（按 created_at，分批删，避免长锁）
    // D-07：两个条件独立清理
    let deletedByAge = 0;
    try {
      const result = await ctx.service.db.query(
        `DELETE FROM gateway_runtime_events
         WHERE created_at < DATE_SUB(NOW(), INTERVAL 1 MONTH)
         LIMIT ?`,
        [DELETE_BATCH_SIZE]
      );
      deletedByAge = (result as { affectedRows?: number })?.affectedRows ?? 0;
      console.info(`[cleanup] deleted ${deletedByAge} runtime_events older than 1 month`);
    } catch (err) {
      console.warn('[cleanup] failed to delete old runtime_events by age:', String(err));
    }

    // 步骤 2：按 session_id 分批删除超过 10 万条的旧行
    // D-07：每个 session 保留最新 MAX_EVENTS_PER_SESSION 条
    let sessionsProcessed = 0;
    let deletedByCount = 0;
    try {
      const sessions = await ctx.service.db.query(
        `SELECT DISTINCT session_id FROM gateway_runtime_events`
      ) as Array<{ session_id: string }>;

      for (const row of sessions) {
        const sessionId = row.session_id;
        try {
          // 找到第 MAX_EVENTS_PER_SESSION 条的 id（倒序第 99999 个，保留 0..99999 共 100000 条）
          const result = await ctx.service.db.query(
            `DELETE FROM gateway_runtime_events
             WHERE session_id = ?
               AND id < (
                 SELECT id FROM (
                   SELECT id FROM gateway_runtime_events
                   WHERE session_id = ?
                   ORDER BY id DESC
                   LIMIT 1 OFFSET ?
                 ) AS t
               )`,
            [sessionId, sessionId, MAX_EVENTS_PER_SESSION - 1]
          );
          const affected = (result as { affectedRows?: number })?.affectedRows ?? 0;
          if (affected > 0) {
            deletedByCount += affected;
            sessionsProcessed++;
          }
        } catch {
          // 该 session 不足 MAX_EVENTS_PER_SESSION 条（OFFSET 超出范围），跳过
        }
      }
      console.info(`[cleanup] deleted ${deletedByCount} runtime_events by count limit across ${sessionsProcessed} sessions`);
    } catch (err) {
      console.warn('[cleanup] failed to delete runtime_events by count limit:', String(err));
    }
  }
}
```
