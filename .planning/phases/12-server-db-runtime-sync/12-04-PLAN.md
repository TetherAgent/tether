```typescript
import { Service } from 'egg';

type GatewaySessionRecord = {
  id: string;
  accountId: string;
  workspaceId: string;
  gatewayId: string;
  userId?: string;
  provider: string;
  title?: string;
  projectPath?: string;
  agentSessionId?: string;
  status: string;
  transport: string;
  lastActiveAt?: number;
  createdAt: number;
  updatedAt: number;
};

type ChatMessageRecord = {
  id: number;
  sessionId: string;
  turnIndex: number;
  role: string;
  content: string;
  toolsJson?: string;
  createdAt: number;
};

type RuntimeEventRecord = {
  id: number;
  sessionId: string;
  eventId: number;
  eventType: string;
  payloadJson: string;
  createdAt: number;
};

export default class SessionRepositoryService extends Service {
  private mysqlModeEnabled() {
    return this.ctx.service.db.mysqlModeEnabled();
  }

  private sqlDateToMs(value: unknown): number {
    if (value instanceof Date) return value.getTime();
    if (typeof value === 'number') return value;
    if (typeof value === 'string') return new Date(value).getTime();
    return Date.now();
  }

  private nullableString(value: unknown): string | undefined {
    return typeof value === 'string' && value ? value : undefined;
  }

  private sessionFromRow(row: Record<string, unknown>): GatewaySessionRecord {
    return {
      id: String(row.id),
      accountId: String(row.account_id),
      workspaceId: String(row.workspace_id),
      gatewayId: String(row.gateway_id),
      userId: this.nullableString(row.user_id),
      provider: String(row.provider),
      title: this.nullableString(row.title),
      projectPath: this.nullableString(row.project_path),
      agentSessionId: this.nullableString(row.agent_session_id),
      status: String(row.status),
      transport: String(row.transport ?? 'pty-event-stream'),
      lastActiveAt: row.last_active_at ? this.sqlDateToMs(row.last_active_at) : undefined,
      createdAt: this.sqlDateToMs(row.created_at),
      updatedAt: this.sqlDateToMs(row.updated_at)
    };
  }

  private chatMessageFromRow(row: Record<string, unknown>): ChatMessageRecord {
    return {
      id: Number(row.id),
      sessionId: String(row.session_id),
      turnIndex: Number(row.turn_index),
      role: String(row.role),
      content: String(row.content),
      toolsJson: this.nullableString(row.tools_json),
      createdAt: this.sqlDateToMs(row.created_at)
    };
  }

  private eventFromRow(row: Record<string, unknown>): RuntimeEventRecord {
    return {
      id: Number(row.id),
      sessionId: String(row.session_id),
      eventId: Number(row.event_id),
      eventType: String(row.event_type),
      payloadJson: String(row.payload_json),
      createdAt: this.sqlDateToMs(row.created_at)
    };
  }

  // GET /api/sessions — 按 account/workspace 过滤，ORDER BY last_active_at DESC
  public async listSessions(
    accountId: string,
    workspaceId: string,
    limit = 50,
    offset = 0
  ): Promise<GatewaySessionRecord[]> {
    if (!this.mysqlModeEnabled()) return [];
    const rows = await this.ctx.service.db.query(
      `SELECT * FROM gateway_sessions
       WHERE account_id = ? AND workspace_id = ?
       ORDER BY last_active_at DESC
       LIMIT ? OFFSET ?`,
      [accountId, workspaceId, limit, offset]
    );
    return (rows as Record<string, unknown>[]).map(row => this.sessionFromRow(row));
  }

  // GET /api/sessions/:id/conversation — 先校验归属，再按 turn_index 升序读取
  public async getConversation(
    sessionId: string,
    accountId: string,
    workspaceId: string
  ): Promise<ChatMessageRecord[]> {
    if (!this.mysqlModeEnabled()) return [];
    // 校验 session 归属（读侧防串）
    const sessionRows = await this.ctx.service.db.query(
      `SELECT id FROM gateway_sessions
       WHERE id = ? AND account_id = ? AND workspace_id = ? LIMIT 1`,
      [sessionId, accountId, workspaceId]
    );
    if (!Array.isArray(sessionRows) || sessionRows.length === 0) return [];
    const rows = await this.ctx.service.db.query(
      `SELECT * FROM gateway_chat_messages
       WHERE session_id = ?
       ORDER BY turn_index ASC`,
      [sessionId]
    );
    return (rows as Record<string, unknown>[]).map(row => this.chatMessageFromRow(row));
  }

  // GET /api/sessions/:id/events — cursor 分页，?limit=&before=
  public async listEvents(
    sessionId: string,
    accountId: string,
    workspaceId: string,
    limit = 100,
    before?: number
  ): Promise<RuntimeEventRecord[]> {
    if (!this.mysqlModeEnabled()) return [];
    // 校验 session 归属（读侧防串）
    const sessionRows = await this.ctx.service.db.query(
      `SELECT id FROM gateway_sessions
       WHERE id = ? AND account_id = ? AND workspace_id = ? LIMIT 1`,
      [sessionId, accountId, workspaceId]
    );
    if (!Array.isArray(sessionRows) || sessionRows.length === 0) return [];
    if (before != null) {
      const rows = await this.ctx.service.db.query(
        `SELECT * FROM gateway_runtime_events
         WHERE session_id = ? AND id < ?
         ORDER BY id DESC
         LIMIT ?`,
        [sessionId, before, limit]
      );
      return (rows as Record<string, unknown>[]).map(row => this.eventFromRow(row));
    }
    const rows = await this.ctx.service.db.query(
      `SELECT * FROM gateway_runtime_events
       WHERE session_id = ?
       ORDER BY id DESC
       LIMIT ?`,
      [sessionId, limit]
    );
    return (rows as Record<string, unknown>[]).map(row => this.eventFromRow(row));
  }
}

import { Controller } from 'egg';

export default class SessionController extends Controller {
  // GET /api/sessions
  public async list(): Promise<void> {
    const { ctx } = this;
    const auth = ctx.state.auth as { accountId?: string; workspaceId?: string } | undefined;
    const accountId = auth?.accountId ?? '';
    const workspaceId = auth?.workspaceId ?? '';
    const limit = Math.min(Number(ctx.query.limit) || 50, 200);
    const offset = Number(ctx.query.offset) || 0;
    const sessions = await ctx.service.sessionRepository.listSessions(accountId, workspaceId, limit, offset);
    ctx.success({ sessions });
  }

  // GET /api/sessions/:id/conversation
  public async conversation(): Promise<void> {
    const { ctx } = this;
    const auth = ctx.state.auth as { accountId?: string; workspaceId?: string } | undefined;
    const accountId = auth?.accountId ?? '';
    const workspaceId = auth?.workspaceId ?? '';
    const sessionId = ctx.params.id;
    const turns = await ctx.service.sessionRepository.getConversation(sessionId, accountId, workspaceId);
    ctx.success({ turns });
  }

  // GET /api/sessions/:id/events
  public async events(): Promise<void> {
    const { ctx } = this;
    const auth = ctx.state.auth as { accountId?: string; workspaceId?: string } | undefined;
    const accountId = auth?.accountId ?? '';
    const workspaceId = auth?.workspaceId ?? '';
    const sessionId = ctx.params.id;
    const limit = Math.min(Number(ctx.query.limit) || 100, 500);
    const before = ctx.query.before ? Number(ctx.query.before) : undefined;
    const events = await ctx.service.sessionRepository.listEvents(sessionId, accountId, workspaceId, limit, before);
    ctx.success({ events });
  }
}

        allow 127.0.0.1;
        deny all;

        proxy_http_version 1.1;

```

**块 2：/api/sessions* — GET 路由到 Server，POST 路由到 Relay（D-10 读写分流）**
```nginx
    # Phase 12: session 读写分流（D-10）
    # GET/HEAD → Server (4800)，读 gateway_chat_messages / gateway_sessions
    # POST → Relay (4889)，input / stop 走 Gateway 反向 RPC
    location ~ ^/api/sessions(/|$) {
        set $sessions_upstream http://127.0.0.1:4889;
        if ($request_method ~ ^(GET|HEAD)$) {
            set $sessions_upstream http://127.0.0.1:4800;
        }
        proxy_pass $sessions_upstream;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection $connection_upgrade;
    }
```
