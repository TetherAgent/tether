    **pty.ts — 实现分层 Map 的 restoreSession（BLOCKER 5 修正）：**

    当前 PtySessionManager 使用 `sessions: Map<string, LivePtySession>` 存储所有 session，其中 LivePtySession 包含真实的 IPty 进程。
    restoreSession 恢复的是无真实 PTY 进程的元数据 session，若混入同一 Map，write/resize 等方法调用 live.pty.write 会崩溃。

    在 PtySessionManager 中新增独立的 `restoredSessions: Map<string, Session>`（仅元数据，无 IPty 进程），并实现 restoreSession：

    ```typescript
    // 在 PtySessionManager 类顶部新增成员变量
    private restoredSessions = new Map<string, Session>();

    /**
     * 从 relay sessions-restore 帧加载仅元数据的 session（无真实 PTY 进程）。
     * 这些 session 不能接受 write / resize 操作，只能被列举和查询。
     */
    restoreSession(session: Session): void {
      // 如果 activeSessions 中已有同 id 的真实 PTY session，不覆盖
      if (this.sessions.has(session.id)) return;
      this.restoredSessions.set(session.id, { ...session });
    }

    // 修改 getSession：先查真实 PTY session，再查恢复的元数据 session
    getSession(id: string): Session | undefined {
      return this.sessions.get(id)?.session ?? this.restoredSessions.get(id);
    }

    // 修改 listSessions：合并两个 Map，真实 PTY session 优先
    listSessions(): Session[] {
      const result = new Map<string, Session>();
      for (const [id, session] of this.restoredSessions) {
        result.set(id, session);
      }
      for (const [id, live] of this.sessions) {
        result.set(id, live.session);  // 真实 PTY session 覆盖恢复的元数据
      }
      return [...result.values()];
    }

    // 修改 updateSessionStatus：两个 Map 都尝试更新
    updateSessionStatus(id: string, status: SessionStatus): void {
      const live = this.sessions.get(id);
      if (live) {
        live.session.status = status;
        return;
      }
      const restored = this.restoredSessions.get(id);
      if (restored) { restored.status = status; }
    }
    ```

       ```typescript
         provider: ProviderDefinition,

         await new Promise<void>((resolve, reject) => {
           ws.once('open', () => resolve());
           ws.once('error', reject);
         });

         // client.auth
         ws.send(JSON.stringify({ type: 'client.auth', token: authToken }));

         const clientId = await new Promise<string>((resolve, reject) => {
           const timer = setTimeout(() => reject(new Error('relay auth timeout')), 5000);
           ws.on('message', (data) => {
             const frame = JSON.parse(String(data));
             if (frame.type === 'client.auth.ok') {
               clearTimeout(timer);
               resolve(frame.clientId as string);
             } else if (frame.type === 'client.auth.failed') {
               clearTimeout(timer);
               reject(new Error(`relay auth failed: ${String(frame.message)}`));
             }
           });
         });

         // client.new-pty-session（含 gatewayId 用于 relay 查找，但 relay 路由依据是 client.gatewayId）
         ws.send(JSON.stringify({
           type: 'client.new-pty-session',
           provider: provider.provider,
           command: provider.command,
           cwd: path.resolve(options.project ?? process.cwd()),
           cols: process.stdout.columns ?? 120,
           rows: process.stdout.rows ?? 40,
           gatewayId
         }));

         const sessionId = await new Promise<string>((resolve, reject) => {
           const timer = setTimeout(() => reject(new Error('session create timeout')), 10000);
           ws.on('message', (data) => {
             const frame = JSON.parse(String(data));
             if (frame.type === 'gateway.session-created') {
               clearTimeout(timer);
               resolve(frame.sessionId as string);
             } else if (frame.type === 'error') {
               clearTimeout(timer);
               reject(new Error(`session create error: ${String(frame.message)}`));
             }
           });
         });

         ws.close();
         return { id: sessionId };
       }
       ```
