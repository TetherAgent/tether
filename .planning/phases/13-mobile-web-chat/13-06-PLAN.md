**阶段 1：HTTP 历史加载（D-43，先 HTTP 再 WS）**：
```typescript
useEffect(() => {
  if (!activeSessionId) return;
  setCurrentSessionId(activeSessionId);

  const load = async () => {
    const { createHttpClient } = await import('../../lib/api.js');
    const http = createHttpClient();
    try {
      const data = await http.get<{ messages: Array<{ role: string; content: string; usageJson?: unknown; createdAt: string }> }>(
        `/api/server/chat-sessions/${activeSessionId}/messages`
      );
      const items: MessageItem[] = (data.messages ?? []).map((m, i) => {
        if (m.role === 'user') {
          return { kind: 'user', id: `h-${i}`, content: m.content, ts: Date.parse(m.createdAt) };
        }
        return { kind: 'agent', id: `h-${i}`, text: m.content, isStreaming: false, isWaiting: false, isLost: false, provider: selectedProvider, usage: m.usageJson as MessageItem['usage'] };
      });

      // 检测 in-flight 状态（最后一条是 user 且无对应 assistant）
      const lastMsg = items.at(-1);
      const isInFlight = lastMsg?.kind === 'user';

      if (isInFlight) {
        // 追加等待气泡
        items.push({ kind: 'agent', id: 'waiting', text: '', isStreaming: false, isWaiting: true, isLost: false, provider: selectedProvider });
        setIsInflight(true);
      }

      setMessages(items);
    } catch {
      // 加载失败显示错误（后续 WS 仍正常建立）
    }
  };

  void load();
}, [activeSessionId]);
```

**阶段 2：WS 连接建立（历史渲染完成后）**：
```typescript
useEffect(() => {
  // WS 连接：复用 session-surface.tsx 的 ws-ticket 逻辑
  const connect = async () => {
    setWsStatus('connecting');
    const { createHttpClient, readStoredNormalAuth } = await import('../../lib/api.js');
    const auth = readStoredNormalAuth();
    if (!auth?.relayUrl) return;

    // 获取 ws-ticket
    const http = createHttpClient();
    let ticket: string;
    try {
      const ticketData = await http.get<{ ticket: string }>('/api/relay/ws-ticket');
      ticket = ticketData.ticket;
    } catch { return; }

    // 构建 WS URL（参考 session-surface.tsx buildRelayClientUrl）
    const url = new URL(auth.relayUrl.trim());
    url.pathname = `${url.pathname.replace(/\/$/, '')}/ws/client`;
    url.search = '';
    url.hash = '';

    const ws = new WebSocket(url.toString());
    wsRef.current = ws;

    ws.onopen = () => {
      setWsStatus('connected');
      // 发送 client.auth
      ws.send(JSON.stringify({ type: 'client.auth', ticket }));
    };

    ws.onclose = () => setWsStatus('disconnected');

    ws.onmessage = (event) => {
      let frame: Record<string, unknown>;
      try { frame = JSON.parse(event.data as string) as Record<string, unknown>; }
      catch { return; }
      handleWsFrame(frame);
    };
  };

  void connect();
  return () => { wsRef.current?.close(); };
}, []);  // 只在 mount/unmount 时连接/断开
```

**WS 帧处理（handleWsFrame）**：
```typescript
const handleWsFrame = (frame: Record<string, unknown>) => {
  const type = frame.type as string;

  switch (type) {
    case 'gateway.session-created': {
      const sid = frame.sessionId as string;
      setCurrentSessionId(sid);
      navigate(`/chats/${sid}`);
      break;
    }
    case 'agent.delta': {
      const text = frame.text as string;
      setMessages(prev => prev.map(m =>
        m.kind === 'agent' && (m.isStreaming || m.isWaiting)
          ? { ...m, text: m.text + text, isStreaming: true, isWaiting: false }
          : m
      ));
      break;
    }
    case 'agent.result': {
      const text = frame.text as string;
      const usage = frame.usage as MessageItem['usage'];
      const durationMs = inflightStartedAtRef.current ? Date.now() - inflightStartedAtRef.current : undefined;
      setMessages(prev => prev.map(m =>
        m.kind === 'agent' && (m.isStreaming || m.isWaiting)
          ? { ...m, text, isStreaming: false, isWaiting: false, usage, durationMs }
          : m
      ));
      setIsInflight(false);
      setTimeout(() => inputRef.current?.focus(), 100);
      break;
    }
    case 'agent.tool': {
      setMessages(prev => [...prev, {
        kind: 'tool', id: `tool-${Date.now()}`,
        toolName: frame.name as string,
        input: frame.input as Record<string, unknown>,
        result: frame.result as string | undefined,
        isError: Boolean(frame.isError),
        isInFlight: frame.result === undefined
      }]);
      break;
    }
    case 'gateway.chat-catchup': {
      const catchupText = frame.text as string;
      setMessages(prev => prev.map(m =>
        m.kind === 'agent' && m.isWaiting
          ? { ...m, text: catchupText, isStreaming: true, isWaiting: false }
          : m
      ));
      break;
    }
    case 'error': {
      if (isInflight) {
        // 当前等待中的气泡标记为 lost
        setMessages(prev => prev.map(m =>
          m.kind === 'agent' && (m.isWaiting || m.isStreaming)
            ? { ...m, isWaiting: false, isStreaming: false, isLost: true }
            : m
        ));
        setIsInflight(false);
      }
      break;
    }
  }
};
```
