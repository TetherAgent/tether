```typescript
function RequireUserAuth({ children }: { children: React.ReactNode }) {
  const location = useLocation();
  const { authReady, normalAuth } = useAuth();
  const storedNormalAuth = normalAuth ?? readStoredNormalAuth();
  if (!authReady) return null;

}
```

**新建 apps/web/src/components/chats/chat-session-list.tsx**：

会话列表组件，HTTP 加载（per D-43 HTTP only for history）：

```typescript
import * as React from 'react';
import { useEffect, useState, useCallback } from 'react';
import { useNavigate } from 'react-router-dom';
import { Search } from 'lucide-react';
import { Input, Skeleton, Empty, Button } from '@tether/design';
import { useI18n } from '../../hooks/use-i18n.js';
import { ModelAvatar } from './model-avatar.js';

export type ChatSession = {
  id: string;
  provider: string;
  projectPath: string;
  status: string;
  lastActiveAt: number;
  createdAt: number;
};

export type ChatSessionListProps = {
  activeSessionId?: string;
  onSelectSession?: (sessionId: string) => void;
};

export function ChatSessionList({ activeSessionId, onSelectSession }: ChatSessionListProps) {
  const navigate = useNavigate();
  const { t } = useI18n();
  const [sessions, setSessions] = useState<ChatSession[]>([]);
  const [loading, setLoading] = useState(true);
  const [searchQuery, setSearchQuery] = useState('');

  const loadSessions = useCallback(async () => {
    try {
      setLoading(true);
      // 从 apps/web/src/lib/api.ts 取 HTTP client
      const { createHttpClient } = await import('../../lib/api.js');
      const http = createHttpClient();
      const data = await http.get<{ sessions: ChatSession[] }>('/api/server/chat-sessions');
      setSessions(data.sessions ?? []);
    } catch {
      setSessions([]);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { void loadSessions(); }, [loadSessions]);

  const handleSelect = (sessionId: string) => {
    onSelectSession?.(sessionId);
    navigate(`/chats/${sessionId}`);
  };

  const filteredSessions = sessions.filter(s =>
    searchQuery === '' || s.projectPath.toLowerCase().includes(searchQuery.toLowerCase())
  );

  return (
    <div className="flex flex-col h-full">
      {/* 搜索框 */}
      <div className="p-2">
        <div className="relative">
          <Search size={14} className="absolute left-2.5 top-1/2 -translate-y-1/2 text-muted-foreground" />
          <Input
            className="pl-8 h-9 text-sm"
            placeholder="⌘K 搜索会话"
            value={searchQuery}
            onChange={e => setSearchQuery(e.target.value)}
          />
        </div>
      </div>

      {/* 会话列表 */}
      <div className="flex-1 overflow-y-auto">
        {loading && (
          <div className="space-y-2 p-2">
            {[1, 2, 3].map(i => <Skeleton key={i} className="h-16 rounded-lg" />)}
          </div>
        )}

        {!loading && filteredSessions.length === 0 && (
          <div className="flex flex-col items-center justify-center h-32 px-4">
            <Empty heading={t.chatsEmptyTitle} body={t.chatsEmptyBody} />
          </div>
        )}

        {!loading && filteredSessions.map(session => (
          <button
            key={session.id}
            className={`w-full flex items-center gap-3 px-3 py-3 min-h-[64px] text-left transition-colors ${
              session.id === activeSessionId
                ? 'bg-sidebar-accent'
                : 'hover:bg-sidebar-accent/60'
            }`}
            onClick={() => handleSelect(session.id)}
          >
            <ModelAvatar provider={session.provider} size={36} />
            <div className="flex-1 min-w-0">
              <div className="flex items-center justify-between gap-1">
                <span className="text-sm font-semibold text-foreground truncate">
                  {session.provider} · {formatDate(session.createdAt)}
                </span>
                {session.status === 'running' && (
                  <span className="shrink-0 w-[7px] h-[7px] rounded-full bg-bull animate-pulse" />
                )}
              </div>
              <p className="text-xs text-muted-foreground truncate mt-0.5">
                {session.projectPath}
              </p>
            </div>
            <span className="shrink-0 text-xs text-foreground-tertiary">
              {formatRelativeTime(session.lastActiveAt, t)}
            </span>
          </button>
        ))}
      </div>
    </div>
  );
}

function formatDate(ts: number): string {
  return new Date(ts).toLocaleDateString('zh-CN', { month: 'numeric', day: 'numeric' });
}

function formatRelativeTime(ts: number, t: ReturnType<typeof useI18n>['t']): string {
  const diff = Date.now() - ts;
  if (diff < 60_000) return t.chatsTimeJustNow;
  if (diff < 3_600_000) return t.chatsTimeMinutes.replace('{n}', String(Math.floor(diff / 60_000)));
  if (diff < 86_400_000) return t.chatsTimeHours.replace('{n}', String(Math.floor(diff / 3_600_000)));
  return t.chatsTimeDays.replace('{n}', String(Math.floor(diff / 86_400_000)));
}
```
