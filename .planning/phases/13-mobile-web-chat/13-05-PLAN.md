```typescript
import * as React from 'react';
import { useState } from 'react';
import ReactMarkdown from 'react-markdown';
import remarkGfm from 'remark-gfm';
import rehypeHighlight from 'rehype-highlight';
import { Copy, Check } from 'lucide-react';
import { StreamingCursor } from './streaming-cursor.js';
import { ThinkingDots } from './thinking-dots.js';
import { ModelAvatar } from './model-avatar.js';

// D-36: 代码块包装器，含复制按钮
function CodeBlockWrapper({ children }: { children: React.ReactNode }) {
  const [copied, setCopied] = useState(false);

  const handleCopy = () => {
    // 从 pre 元素提取文本
    const codeText = (children as React.ReactElement)?.props?.children?.props?.children ?? '';
    void navigator.clipboard.writeText(String(codeText)).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    });
  };

  return (
    <div className="relative group my-2">
      <pre className="text-xs font-mono leading-relaxed overflow-x-auto p-3 rounded-md"
           style={{ background: 'var(--code-bg, #f1f5f9)', color: 'var(--code-text, #0f172a)' }}>
        {children}
      </pre>
      <button
        className="absolute top-2 right-2 w-7 h-7 flex items-center justify-center rounded opacity-0 group-hover:opacity-100 transition-opacity bg-background/80 hover:bg-background"
        onClick={handleCopy}
        aria-label="Copy code"
      >
        {copied ? <Check size={12} className="text-green-500" /> : <Copy size={12} className="text-muted-foreground" />}
      </button>
    </div>
  );
}

// D-38: 检测并补齐未闭合 code fence
function closeUnclosedFence(text: string): string {
  const fenceCount = (text.match(/^```/gm) ?? []).length;
  return fenceCount % 2 === 1 ? text + '\n```' : text;
}

export type ChatBubbleAgentProps = {
  text: string;
  isStreaming: boolean;
  /** 纯等待状态（无文字，显示 ThinkingDots） */
  isWaiting?: boolean;
  provider?: string;
  resultCard?: React.ReactNode;
};

export function ChatBubbleAgent({ text, isStreaming, isWaiting, provider, resultCard }: ChatBubbleAgentProps) {
  const renderText = isStreaming ? closeUnclosedFence(text) : text;

  return (
    <div className="flex items-start gap-2 mb-3">
      <ModelAvatar provider={provider ?? 'claude'} size={32} />
      <div className="flex-1 min-w-0">
        <div
          className="rounded-2xl rounded-tl-sm px-4 py-2 text-sm leading-[1.55]"
          style={{ background: 'var(--agent-bubble, #fff)' }}
        >
          {isWaiting ? (
            <ThinkingDots />
          ) : (
            <>
              <ReactMarkdown
                remarkPlugins={[remarkGfm]}
                rehypePlugins={[rehypeHighlight]}
                components={{
                  pre: ({ children }) => <CodeBlockWrapper>{children}</CodeBlockWrapper>,
                  // D-36: 链接 target=_blank
                  a: ({ href, children }) => (
                    <a
                      href={href}
                      target="_blank"
                      rel="noopener noreferrer"
                      className="text-blue-600 dark:text-blue-400 underline"
                    >
                      {children}
                    </a>
                  ),
                  // D-36: 图片渲染为 alt 文本
                  img: ({ alt }) => (
                    <span className="text-muted-foreground italic">[image: {alt ?? ''}]</span>
                  ),
                  // 段落间距
                  p: ({ children }) => <p className="mb-2 last:mb-0">{children}</p>,
                  // 内联代码
                  code: ({ children, className }) => {
                    const isBlock = className?.includes('language-');
                    if (isBlock) return <code className={className}>{children}</code>;
                    return (
                      <code
                        className="text-xs font-mono px-1 rounded-sm"
                        style={{ background: 'var(--code-bg, #f1f5f9)', color: 'var(--code-text, #0f172a)' }}
                      >
                        {children}
                      </code>
                    );
                  }
                }}
              >
                {renderText}
              </ReactMarkdown>
              {isStreaming && <StreamingCursor />}
            </>
          )}
        </div>
        {resultCard && (
          <div className="mt-1">{resultCard}</div>
        )}
      </div>
    </div>
  );
}
```

import * as React from 'react';
import { ChevronRight } from 'lucide-react';
import {
  Collapsible, CollapsibleContent, CollapsibleTrigger
} from '@tether/design';

export type ToolCardProps = {
  toolName: string;
  input: Record<string, unknown>;
  result?: string;
  isError?: boolean;
  isInFlight?: boolean;
};

export function ToolCard({ toolName, input, result, isError, isInFlight }: ToolCardProps) {
  const badgeStyle = isError
    ? { background: 'rgba(220,38,38,0.12)', color: 'var(--error-accent, #dc2626)' }
    : { background: 'rgba(99,102,241,0.12)', color: 'var(--tool-accent, #6366f1)' };

  const inlineArgs = JSON.stringify(input).slice(0, 60);
  const hasMore = JSON.stringify(input).length > 60;

  return (
    <Collapsible className="rounded-lg my-2 overflow-hidden" style={{ border: '1px solid var(--tool-border, #e2e8f0)', background: 'var(--tool-bg, #f8fafc)' }}>
      <CollapsibleTrigger className="flex items-center gap-2 w-full px-3 py-2 text-left hover:bg-black/[0.02]">
        <span
          className="text-2xs font-semibold font-mono px-1.5 py-0.5 rounded-sm uppercase tracking-wide"
          style={badgeStyle}
        >
          {toolName}
        </span>
        <span className="text-xs text-muted-foreground font-mono truncate flex-1">
          {inlineArgs}{hasMore ? '…' : ''}
        </span>
        <ChevronRight size={14} className="text-muted-foreground shrink-0 transition-transform duration-150 data-[state=open]:rotate-90" />
      </CollapsibleTrigger>
      <CollapsibleContent>
        <div className="border-t border-dashed px-3 py-2 text-xs font-mono" style={{ borderColor: 'var(--tool-border, #e2e8f0)' }}>
          <pre className="whitespace-pre-wrap break-all">{JSON.stringify(input, null, 2)}</pre>
        </div>
        {isInFlight && (
          <div className="px-3 py-2 text-muted-foreground text-xs italic">running…</div>
        )}
        {!isInFlight && result && (
          <div
            className="rounded-md mx-2 mb-2 p-2 max-h-44 overflow-y-auto"
            style={{
              background: isError ? 'var(--error-bg, #fef2f2)' : 'var(--tool-result-bg, #ecfdf5)',

            <p
              className="text-2xs font-semibold uppercase tracking-widest mb-1"
              style={{ color: isError ? 'var(--error-accent, #dc2626)' : 'var(--tool-result-accent, #059669)' }}
            >
              {isError ? 'ERROR' : 'RESULT'}
            </p>
            <pre className="text-xs font-mono whitespace-pre-wrap break-all">{result}</pre>
          </div>
        )}
      </CollapsibleContent>
    </Collapsible>
  );
}
